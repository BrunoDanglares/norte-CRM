TAREFA 1 — Criar utilitário de cache de avatar
Crie o arquivo server/services/avatarCache.ts:

```typescript
import fs from "fs";
import path from "path";
import https from "https";
import http from "http";

const uploadsDir = path.resolve("uploads");

export async function downloadAndCacheAvatar(
  externalUrl: string,
  workspaceId: string,
  phone: string
): Promise<string | null> {
  try {
    const dir = path.join(uploadsDir, "avatars", workspaceId);
    if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });

    const filename = `${phone.replace(/\D/g, "")}.jpg`;
    const localPath = path.join(dir, filename);
    const publicUrl = `/uploads/avatars/${workspaceId}/${filename}`;

    await new Promise<void>((resolve, reject) => {
      const protocol = externalUrl.startsWith("https") ? https : http;
      const file = fs.createWriteStream(localPath);
      protocol.get(externalUrl, (res) => {
        if (res.statusCode !== 200) {
          file.close();
          fs.unlink(localPath, () => {});
          reject(new Error(`HTTP ${res.statusCode}`));
          return;
        }
        res.pipe(file);
        file.on("finish", () => { file.close(); resolve(); });
      }).on("error", (err) => {
        file.close();
        fs.unlink(localPath, () => {});
        reject(err);
      });
    });

    return publicUrl;
  } catch (err) {
    console.error("[avatarCache] Erro ao baixar avatar:", err);
    return null;
  }
}

export function localAvatarExists(workspaceId: string, phone: string): string | null {
  const filename = `${phone.replace(/\D/g, "")}.jpg`;
  const localPath = path.join(uploadsDir, "avatars", workspaceId, filename);
  if (fs.existsSync(localPath)) {
    return `/uploads/avatars/${workspaceId}/${filename}`;
  }
  return null;
}
```

TAREFA 3 — Backfill das URLs expiradas existentes

Crie a função backfillExpiredAvatars() em server/services/avatarCache.ts:

```typescript
import { db } from "../db"; // ajuste o import conforme o projeto
import { conversations } from "../../shared/schema";
import { isNull, not, like } from "drizzle-orm";

export async function backfillExpiredAvatars() {
  try {
    // Busca conversas com URL externa (não local) ainda salva
    const rows = await db
      .select({ id: conversations.id, avatar: conversations.avatar, 
                workspaceId: conversations.workspaceId, telefone: conversations.telefone })
      .from(conversations)
      .where(like(conversations.avatar, "http%"));

    console.log(`[avatarCache] Backfill: ${rows.length} conversas com URL externa`);

    for (const row of rows) {
      if (!row.avatar || !row.workspaceId || !row.telefone) continue;
      const localUrl = await downloadAndCacheAvatar(row.avatar, row.workspaceId, row.telefone);
      if (localUrl) {
        await db.update(conversations)
          .set({ avatar: localUrl })
          .where(eq(conversations.id, row.id));
        console.log(`[avatarCache] Cacheado: ${row.telefone} → ${localUrl}`);
      }
    }
  } catch (err) {
    console.error("[avatarCache] Erro no backfill:", err);
  }
}
```
