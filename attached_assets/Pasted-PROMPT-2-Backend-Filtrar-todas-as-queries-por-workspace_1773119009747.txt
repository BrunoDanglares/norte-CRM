## PARTE 3: Corrigir as rotas para passar workspaceId

Em cada rota que chama métodos do storage, passe o workspaceId:
```typescript// Exemplo leads
router.get('/api/leads', requireAuth, async (req: AuthRequest, res) => {
const workspaceId = getWorkspaceId(req)
const leads = await storage.getLeads(workspaceId)
res.json(leads)
})router.post('/api/leads', requireAuth, async (req: AuthRequest, res) => {
const workspaceId = getWorkspaceId(req)
const lead = await storage.createLead({ ...req.body, workspace_id: workspaceId })
res.json(lead)
})router.patch('/api/leads/:id', requireAuth, async (req: AuthRequest, res) => {
const workspaceId = getWorkspaceId(req)
const lead = await storage.updateLead(req.params.id, req.body, workspaceId)
if (!lead) return res.status(404).json({ error: 'Lead não encontrado' })
res.json(lead)
})

Aplique para todas as rotas de todas as entidades listadas na Parte 2.
