// ── Serviço ──────────────────────────────────────────────────────────────────

export class N8nAiService {

  // Busca template do DB. Se não encontrar, usa o default.
  private async getTemplate(workspaceId: string, name: string) {
    const rows = await db
      .select()
      .from(ai_prompt_templates)
      .where(
        and(
          eq(ai_prompt_templates.workspace_id, workspaceId),
          eq(ai_prompt_templates.name, name),
          eq(ai_prompt_templates.is_active, true)
        )
      )
      .limit(1);

    if (rows[0]) return rows[0].system_prompt;

    // Fallback para template padrão
    const def = DEFAULT_TEMPLATES[name];
    if (def) return def.system_prompt;

    throw new Error(`Template '${name}' não encontrado e sem default disponível`);
  }

  // ── GERAR RESPOSTA IA ──────────────────────────────────────────────────────
  async gerarResposta(params: {
    workspaceId: string;
    prompt_name: string;
    context: Record<string, any>;       // dados ERP + score + promises
    mensagem: string;                   // mensagem atual do cliente
    historico?: Array<{ role: 'user' | 'assistant'; content: string }>;
  }): Promise<{ response: string; prompt_name: string; tokens_used: number }> {

    const systemPromptBase = await this.getTemplate(params.workspaceId, params.prompt_name);

    // Injeta contexto estruturado no system prompt
    const contextBlock = `
---
DADOS REAIS DO CLIENTE (use exclusivamente esses dados, nunca invente):
${JSON.stringify(params.context, null, 2)}

Data/hora atual: ${new Date().toLocaleString('pt-BR', { timeZone: 'America/Sao_Paulo', weekday: 'long', day: 'numeric', month: 'long', year: 'numeric', hour: '2-digit', minute: '2-digit' })}
---`;

    const systemPrompt = systemPromptBase + contextBlock;

    // Monta histórico de mensagens
    const messages: OpenAI.Chat.Completions.ChatCompletionMessageParam[] = [
      { role: 'system', content: systemPrompt },
      ...(params.historico ?? []),
      { role: 'user', content: params.mensagem },
    ];

    const completion = await openai.chat.completions.create({
      model:       'gpt-4o-mini',
      messages,
      max_tokens:  1000,
      temperature: 0.7,
    });

    const response   = completion.choices[0]?.message?.content ?? '';
    const tokens_used = completion.usage?.total_tokens ?? 0;

    return { response, prompt_name: params.prompt_name, tokens_used };
  }

  // ── SEED DE TEMPLATES PADRÃO ───────────────────────────────────────────────
  // Cria os templates ISP para o workspace se ainda não existirem
  async seedDefaultTemplates(workspaceId: string): Promise<{
    created: string[];
    skipped: string[];
  }> {
    const created: string[] = [];
    const skipped: string[] = [];

    for (const [name, template] of Object.entries(DEFAULT_TEMPLATES)) {
      // Verifica se já existe
      const existing = await db
        .select({ id: ai_prompt_templates.id })
        .from(ai_prompt_templates)
        .where(
          and(
            eq(ai_prompt_templates.workspace_id, workspaceId),
            eq(ai_prompt_templates.name, name)
          )
        )
        .limit(1);

      if (existing.length > 0) {
        skipped.push(name);
        continue;
      }

      await db.insert(ai_prompt_templates).values({
        workspace_id:  workspaceId,
        name,
        description:   template.description,
        system_prompt: template.system_prompt,
        is_active:     true,
      });

      created.push(name);
    }

    return { created, skipped };
  }

  // ── LISTAR TEMPLATES DO WORKSPACE ─────────────────────────────────────────
  async listarTemplates(workspaceId: string) {
    return db
      .select({
        id:          ai_prompt_templates.id,
        name:        ai_prompt_templates.name,
        description: ai_prompt_templates.description,
        is_active:   ai_prompt_templates.is_active,
        updated_at:  ai_prompt_templates.updated_at,
      })
      .from(ai_prompt_templates)
      .where(eq(ai_prompt_templates.workspace_id, workspaceId))
      .orderBy(ai_prompt_templates.name);
  }

  // ── SALVAR/ATUALIZAR TEMPLATE ──────────────────────────────────────────────
  async salvarTemplate(workspaceId: string, params: {
    name: string;
    system_prompt: string;
    description?: string;
  }) {
    const existing = await db
      .select({ id: ai_prompt_templates.id })
      .from(ai_prompt_templates)
      .where(
        and(
          eq(ai_prompt_templates.workspace_id, workspaceId),
          eq(ai_prompt_templates.name, params.name)
        )
      )
      .limit(1);

    if (existing.length > 0) {
      await db
        .update(ai_prompt_templates)
        .set({
          system_prompt: params.system_prompt,
          description:   params.description,
          updated_at:    new Date(),
        })
        .where(
          and(
            eq(ai_prompt_templates.workspace_id, workspaceId),
            eq(ai_prompt_templates.name, params.name)
          )
        );
      return { action: 'updated', name: params.name };
    }

    await db.insert(ai_prompt_templates).values({
      workspace_id:  workspaceId,
      name:          params.name,
      description:   params.description,
      system_prompt: params.system_prompt,
    });
    return { action: 'created', name: params.name };
  }
}

export const n8nAiService = new N8nAiService();

import { n8nAiService } from '../services/n8nAiService';

// ─────────────────────────────────────────────────────────────────────────────
// POST /api/n8n/ai
// Body: { prompt_name, context, mensagem, historico? }
// ─────────────────────────────────────────────────────────────────────────────
router.post('/ai', async (req: Request, res: Response) => {
  try {
    const workspaceId = resolveWorkspaceId(req);
    const { prompt_name, context, mensagem, historico } = req.body as {
      prompt_name: string;
      context:     Record<string, any>;
      mensagem:    string;
      historico?:  Array<{ role: 'user' | 'assistant'; content: string }>;
    };

    if (!prompt_name || !mensagem) {
      return res.status(400).json({
        error: 'Campos obrigatórios: prompt_name, mensagem'
      });
    }

    const result = await n8nAiService.gerarResposta({
      workspaceId,
      prompt_name,
      context:  context ?? {},
      mensagem,
      historico,
    });

    return res.json({ ok: true, ...result });

  } catch (err: any) {
    console.error('[N8N AI]', err);
    return res.status(500).json({ ok: false, error: err.message });
  }
});

// ─────────────────────────────────────────────────────────────────────────────
// GET /api/n8n/ai/templates
// Lista todos os templates do workspace
// ─────────────────────────────────────────────────────────────────────────────
router.get('/ai/templates', async (req: Request, res: Response) => {
  try {
    const workspaceId = resolveWorkspaceId(req);
    const templates = await n8nAiService.listarTemplates(workspaceId);
    return res.json({ ok: true, templates });
  } catch (err: any) {
    return res.status(500).json({ ok: false, error: err.message });
  }
});

// ─────────────────────────────────────────────────────────────────────────────
// PUT /api/n8n/ai/templates/:name
// Salva ou atualiza um template
// ─────────────────────────────────────────────────────────────────────────────
router.put('/ai/templates/:name', async (req: Request, res: Response) => {
  try {
    const workspaceId = resolveWorkspaceId(req);
    const { name } = req.params;
    const { system_prompt, description } = req.body;

    if (!system_prompt) {
      return res.status(400).json({ error: 'Campo system_prompt é obrigatório' });
    }

    const result = await n8nAiService.salvarTemplate(workspaceId, {
      name,
      system_prompt,
      description,
    });

    return res.json({ ok: true, ...result });

  } catch (err: any) {
    return res.status(500).json({ ok: false, error: err.message });
  }
});
