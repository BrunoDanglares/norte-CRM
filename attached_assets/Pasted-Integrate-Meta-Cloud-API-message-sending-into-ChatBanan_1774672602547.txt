Integrate Meta Cloud API message sending into ChatBanana's existing message sending infrastructure, and add channel selection logic so each workspace routes messages through either Z-API or Meta Cloud API.

## Context
- ChatBanana currently sends messages via Z-API
- After this change, the platform checks which channel the workspace uses and routes accordingly
- The automation engine sends messages — we need it to use the right channel per workspace
- Add a message sending abstraction layer (channel router)

export interface SendMessageParams {
  workspaceId: string;
  to: string;                     // phone number
  type: 'text' | 'image' | 'audio' | 'document' | 'video' | 'template';
  content?: string;               // text body
  mediaUrl?: string;              // for media messages
  mediaCaption?: string;
  filename?: string;              // for documents
  templateName?: string;          // for template messages
  templateLanguage?: string;
  templateComponents?: any[];
  replyToMessageId?: string;      // quote a message
  conversationId?: number;        // for logging
}

export interface SendMessageResult {
  success: boolean;
  messageId?: string;             // platform message ID (wamid or z-api id)
  channel: 'zapi' | 'meta';
  error?: string;
}

export async function sendMessage(params: SendMessageParams): Promise<SendMessageResult> {
  const { workspaceId } = params;
  
  // 1. Check if workspace has active Meta Cloud API connection
  const metaConnection = await db.query.whatsapp_official_connections.findFirst({
    where: and(
      eq(whatsapp_official_connections.workspace_id, workspaceId),
      eq(whatsapp_official_connections.status, 'active')
    )
  });
  
  if (metaConnection) {
    return sendViaMetaCloudAPI(params, metaConnection);
  }
  
  // 2. Fallback to Z-API
  return sendViaZAPI(params, workspaceId);
}

async function sendViaMetaCloudAPI(params: SendMessageParams, connection: SelectWhatsappOfficialConnection): Promise<SendMessageResult> {
  try {
    const { phoneNumberId, access_token } = connection;
    let messageId: string;
    
    switch (params.type) {
      case 'text':
        const result = await metaWhatsApp.sendTextMessage({
          phoneNumberId,
          accessToken: access_token,
          to: params.to,
          text: params.content || '',
          replyToMessageId: params.replyToMessageId
        });
        messageId = result.messageId;
        break;
        
      case 'image':
      case 'audio':
      case 'document':
      case 'video':
        const mediaResult = await metaWhatsApp.sendMediaMessage({
          phoneNumberId,
          accessToken: access_token,
          to: params.to,
          type: params.type,
          mediaUrl: params.mediaUrl!,
          caption: params.mediaCaption,
          filename: params.filename
        });
        messageId = mediaResult.messageId;
        break;
        
      case 'template':
        const tplResult = await metaWhatsApp.sendTemplateMessage({
          phoneNumberId,
          accessToken: access_token,
          to: params.to,
          templateName: params.templateName!,
          language: params.templateLanguage || 'pt_BR',
          components: params.templateComponents || []
        });
        messageId = tplResult.messageId;
        break;
        
      default:
        throw new Error(`Unsupported message type: ${params.type}`);
    }
    
    // Update last_used_at
    await db.update(whatsapp_official_connections)
      .set({ last_used_at: new Date() })
      .where(eq(whatsapp_official_connections.id, connection.id));
    
    return { success: true, messageId, channel: 'meta' };
    
  } catch (err: any) {
    console.error('[ChannelRouter] Meta send failed:', err.message);
    return { success: false, error: err.message, channel: 'meta' };
  }
}

## Helper: buildTemplateComponents
```typescript
function buildTemplateComponents(variableConfig: Record<string, string>, contextVars: Record<string, any>) {
  // variableConfig: { "1": "{{nome}}", "2": "{{valor}}" }
  // Resolves placeholders from context and builds Meta's components array
  const bodyParams = Object.entries(variableConfig).map(([_, tpl]) => ({
    type: 'text',
    text: tpl.replace(/\{\{(\w+)\}\}/g, (_, k) => contextVars[k] ?? '')
  }));
  
  return [{ type: 'body', parameters: bodyParams }];
}
```
