  async updateSession(conversationId: string, updates: any) {
    try {
      const fields = {
        fluxo_atual:      updates.fluxoAtual,
        cliente_nome:     updates.clienteNome,
        cliente_id_erp:   updates.clienteIdErp,
        identificado:     updates.identificado,
        ultimo_intent:    updates.ultimoIntent,
        cliente_contrato: updates.clienteContrato,
      };

      for (const [key, value] of Object.entries(fields)) {
        if (value === undefined || value === null) continue;
        await db.execute(
          sql`UPDATE isp_session_state
              SET ${sql.raw(key)} = ${value}, updated_at = now()
              WHERE conversation_id = ${conversationId}`
        );
      }

      if (updates.dadosColetados) {
        await db.execute(
          sql`UPDATE isp_session_state
              SET dados_coletados = dados_coletados || ${JSON.stringify(updates.dadosColetados)}::jsonb,
                  updated_at = now()
              WHERE conversation_id = ${conversationId}`
        );
      }
    } catch (err: any) {
      console.error('[Memory] updateSession error:', err.message);
    }
  }

No arquivo server/services/n8nManagerService.ts, adicione um método
tornarSaveNaoCritico que adiciona continueOnFail: true em todos os

  async tornarSaveNaoCritico(workflowId: string): Promise<any> {
    const workflow = await n8nFetch(`/workflows/${workflowId}`);

    const updatedNodes = workflow.nodes.map((node: any) => {
      const url = node.parameters?.url ?? '';
      if (url.includes('/memory/save') || url.includes('/memory/turn')) {
        return { ...node, continueOnFail: true, alwaysOutputData: true };
      }
      return node;
    });

    await this.atualizarWorkflow(workflowId, {
      name: workflow.name, nodes: updatedNodes,
      connections: workflow.connections,
      settings: { executionOrder: workflow.settings?.executionOrder ?? 'v1' },
    });

    return { ok: true };
  }
