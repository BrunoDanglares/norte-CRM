Implemente a função auxiliar mapFluxoToCapabilityKey:

  function mapFluxoToCapabilityKey(fluxo: string): string {
    const map: Record<string, string> = {
      financeiro:      'FINANCEIRO',
      suporte:         'SUPORTE_TECNICO',
      comercial:       'VENDAS',
      cancelamento:    'CANCELAMENTO',
      financeiroAgent: 'FINANCEIRO',
      suporteAgent:    'SUPORTE_TECNICO',
      comercialAgent:  'VENDAS',
    };
    return map[fluxo] ?? fluxo.toUpperCase();
  }

E getTenantCapabilities:

  async function getTenantCapabilities(
    workspaceId: string
  ): Promise<AgentCapabilities | undefined> {
    const settings = await tenantSettingsService.getByWorkspace(workspaceId);
    return settings?.businessRules?.agent_capabilities;
  }

══ PASSO 4 — Check no resolveIntent ══

Em server/services/agents/priorityResolver.ts, atualize
a função resolveIntent para aceitar capabilities:

  export function resolveIntent(
    vector: IntentVector,
    tenantPriorities?: Record<string, number>,
    capabilities?: AgentCapabilities
  ): string {
    const priorities = {
      ...DEFAULT_PRIORITIES,
      ...(tenantPriorities ?? {}),
    };

    const candidates = vector.intents
      .filter(i => i.confidence >= 0.3)
      .filter(i => {
        // Excluir intents cujo agente está desativado
        if (!capabilities) return true;
        const key = i.intent as keyof AgentCapabilities;
        const agentIntent = key === 'SUPORTE_TECNICO'
          ? 'SUPORTE_TECNICO' : key;
        return capabilities[agentIntent]?.enabled !== false;
      })
      .sort((a, b) => {
        const pa = priorities[a.intent] ?? 99;
        const pb = priorities[b.intent] ?? 99;
        if (pa !== pb) return pa - pb;
        return b.confidence - a.confidence;
      });

    // Se todos os candidatos foram filtrados,
    // fallback para HUMANO (agente sempre disponível)
    if (candidates.length === 0) return 'HUMANO';

    return candidates[0]?.intent ?? vector.dominant;
  }
