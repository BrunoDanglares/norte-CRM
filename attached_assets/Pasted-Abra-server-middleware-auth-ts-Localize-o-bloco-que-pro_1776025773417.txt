CORREÇÃO — validar que o workspace_id enviado existe no banco:

Substitua o bloco pelo código abaixo:

  const serviceToken = req.headers['x-n8n-token'] || req.headers['x-service-token'];
  if (serviceToken && serviceToken === process.env.N8N_SERVICE_TOKEN) {
    const workspaceId = req.body?.workspace_id || req.query?.workspace_id;

    if (!workspaceId) {
      return res.status(400).json({ error: 'workspace_id obrigatório para service token' });
    }

    // Valida que o workspace existe no banco
    const workspace = await db
      .select({ id: workspaces.id })
      .from(workspaces)
      .where(eq(workspaces.id, workspaceId))
      .limit(1);

    if (!workspace.length) {
      console.warn('[auth] service token com workspace_id inválido:', workspaceId);
      return res.status(401).json({ error: 'workspace_id inválido' });
    }

    (req as any).user = {
      id: 0,
      role: 'service',
      workspaceId,
      nome: 'Service Token',
    };
    return next();
  }
