No arquivo server/routes/n8n.ts, adicione esta rota antes
do export default router:

router.post('/manager/deploy-isp-v5', async (req: Request, res: Response) => {
  try {
    const fs = await import('fs');
    const path = await import('path');

    // Lê o JSON do V5 — faça upload do arquivo ISP-Agent-V5-Memory.json
    // na raiz do projeto Replit antes de executar
    const wfPath = path.join(process.cwd(), 'ISP-Agent-V5-Memory.json');
    if (!fs.existsSync(wfPath)) {
      return res.status(404).json({
        error: 'Arquivo ISP-Agent-V5-Memory.json não encontrado na raiz do projeto. Faça upload primeiro.'
      });
    }

    const workflow = JSON.parse(fs.readFileSync(wfPath, 'utf8'));
    const result   = await n8nManagerService.deploy(workflow);
    return res.json({ ok: true, ...result });
  } catch (err: any) {
    return res.status(500).json({ ok: false, error: err.message });
  }
});
