import { z } from 'zod';

const patchConversationSchema = z.object({
  status: z.enum(['open', 'pending', 'resolved']).optional(),
  prioridade: z.enum(['alta', 'media', 'baixa']).optional(),
  pipeline: z.string().max(100).optional(),
  pipeline_etapa: z.string().max(100).optional(),
  assigned_user_id: z.number().nullable().optional(),
  assigned_user_name: z.string().max(100).nullable().optional(),
  assigned_team_id: z.string().uuid().nullable().optional(),
  pendente: z.boolean().optional(),
  tags: z.array(z.string().max(50)).max(20).optional(),
  ai_paused: z.boolean().optional(),
  agente: z.string().max(100).nullable().optional(),
}).strict(); // .strict() rejeita campos não listados

PASSO 2 — No handler, substitua o uso direto de req.body por:

  const parsed = patchConversationSchema.safeParse(req.body);
  if (!parsed.success) {
    return res.status(400).json({
      error: 'Dados inválidos',

  }

ATENÇÃO: Se o schema precisar de campos adicionais que já existem

para ver quais campos ele aceita e inclua todos no schema.
Não use .strict() se isso quebrar chamadas existentes — nesse caso
use .strip() (padrão do Zod) que ignora campos extras silenciosamente.
