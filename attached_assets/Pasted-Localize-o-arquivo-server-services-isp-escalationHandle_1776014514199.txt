export async function assignTeamByAgent(
  conversationId: number,
  workspaceId: string,
  situation: string
): Promise<void> {
  const teamName = SITUATION_TEAM_MAP[situation];
  if (!teamName) return;

  try {
    // Busca o team_id real pelo nome da equipe no workspace
    const team = await db
      .select({ id: teams.id })
      .from(teams)
      .where(and(eq(teams.workspaceId, workspaceId), eq(teams.nome, teamName)))
      .limit(1);

    if (!team.length) return;

    const teamId = team[0].id;

    // Verifica se já está atribuída à mesma equipe (evita update e broadcast desnecessários)
    const current = await db
      .select({ assignedTeamId: conversations.assignedTeamId })
      .from(conversations)
      .where(eq(conversations.id, conversationId))
      .limit(1);

    if (current[0]?.assignedTeamId === teamId) return;

    // Atualiza assigned_team_id e pipeline
    await db
      .update(conversations)
      .set({ assignedTeamId: teamId, pipeline: teamName, updatedAt: new Date() })
      .where(eq(conversations.id, conversationId));

    // Importa broadcastToWorkspace dinamicamente (padrão já usado no arquivo)
    const { broadcastToWorkspace } = await import('../index');
    broadcastToWorkspace(workspaceId, 'conversation_updated', {
      conversationId,
      assigned_team_id: teamId,
      team_name: teamName,
      pipeline: teamName,
    });
  } catch (err) {
    console.error('[assignTeamByAgent]', err);
  }
}

IMPORTANTE: Use os mesmos imports de db, tables (teams, conversations) e
operadores (eq, and) que já existem no arquivo. Não adicione novos imports
que ainda não estejam presentes — só adapte para o padrão Drizzle do projeto.
