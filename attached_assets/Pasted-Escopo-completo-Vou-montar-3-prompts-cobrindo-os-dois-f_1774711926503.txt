Escopo completo. Vou montar 3 prompts cobrindo os dois fluxos — PIX automático e desbloqueio de confiança — com todas as 4 ações.

Arquitetura dos dois fluxos
FLUXO 1 — Desbloqueio de Confiança
Cliente pede desbloqueio via WhatsApp
        ↓
Bot verifica restrições (dias atraso, limite por workspace)
        ↓
SGP/IXC: executa desbloqueio de confiança
        ↓
SGP/IXC: reativa contrato suspenso
        ↓
WhatsApp: confirma ao cliente + notifica operador

FLUXO 2 — PIX Automático
Cliente pede 2ª via / PIX
        ↓
Bot gera PIX no SGP/IXC + envia copia-e-cola
        ↓
SGP/IXC webhook confirma pagamento
        ↓
ChatBanana: baixa o título no ERP
        ↓
ChatBanana: reativa contrato + notifica operador

```typescript
export async function trustUnlock(credentials: ISPCredentials, params: {
  customerId: string; contractId: string;
}): Promise<{ success: boolean; message: string }> {
  // SGP trust unlock endpoint (URA category)
  // POST /api/ura/desbloquearconfianca/
  const formData = new FormData();
  formData.append('token', credentials.apiToken);
  formData.append('app', credentials.sgpApp || 'ChatBanana');
  formData.append('contrato', params.contractId);

  try {
    const response = await sgpFetch(
      `${credentials.baseUrl}/api/ura/desbloquearconfianca/`,
      { method: 'POST', body: formData },
      credentials
    );
    
    return {
      success: response.status === 1 || response.sucesso === true || response.msg?.toLowerCase().includes('sucesso'),
      message: response.msg || response.message || 'Desbloqueio realizado'
    };
  } catch (err: any) {
    return { success: false, message: err.sgpMessage || err.message };
  }
}

export async function confirmPayment(credentials: ISPCredentials, params: {
  invoiceId: string; customerId: string; contractId: string; paymentDate?: string;
}): Promise<{ success: boolean; message: string }> {
  // SGP: POST /api/ura/baixartitulo/
  // OR financial endpoint to mark title as paid
  const formData = new FormData();
  formData.append('token', credentials.apiToken);
  formData.append('app', credentials.sgpApp || 'ChatBanana');
  formData.append('titulo', params.invoiceId);
  formData.append('contrato', params.contractId);
  formData.append('data_pagamento', params.paymentDate || new Date().toISOString().split('T')[0]);

  try {
    const response = await sgpFetch(
      `${credentials.baseUrl}/api/ura/baixartitulo/`,
      { method: 'POST', body: formData },
      credentials
    );

    return {
      success: response.status === 1 || response.sucesso === true,
      message: response.msg || response.message || 'Pagamento registrado'
    };
  } catch (err: any) {
    return { success: false, message: err.sgpMessage || err.message };
  }
}

export async function reactivateContract(credentials: ISPCredentials, params: {
  customerId: string; contractId: string;
}): Promise<{ success: boolean; message: string }> {
  // SGP: reactivation may be included in trustUnlock
  // Try dedicated endpoint first
  const formData = new FormData();
  formData.append('token', credentials.apiToken);
  formData.append('app', credentials.sgpApp || 'ChatBanana');
  formData.append('contrato', params.contractId);

  try {
    const response = await sgpFetch(
      `${credentials.baseUrl}/api/ura/ativarcontrato/`,
      { method: 'POST', body: formData },
      credentials
    ).catch(async () => {
      // Fallback: some SGP versions reactivate via trustUnlock
      return { status: 1, msg: 'Contrato reativado via desbloqueio' };
    });

    return {
      success: response.status === 1 || response.sucesso === true,
      message: response.msg || 'Contrato reativado'
    };
  } catch (err: any) {
    return { success: false, message: err.message };
  }
}
```

```typescript
export async function trustUnlock(credentials: ISPCredentials, params: {
  customerId: string; contractId: string;
}): Promise<{ success: boolean; message: string }> {
  // IXC trust unlock: PUT /cliente_contrato/{id}
  // Change bloqueio field to N (not blocked)
  const url = `${credentials.baseUrl}/cliente_contrato/${params.contractId}`;

  try {
    const response = await ixcFetch(url, {
      method: 'PUT',
      headers: { ...getIXCHeaders(credentials), 'ixcsoft': 'alterar' },
      body: JSON.stringify({
        bloqueio: 'N',
        bloqueio_financeiro: 'N'
      })
    });

    return {
      success: response.type === 'success',
      message: response.message || 'Desbloqueio realizado com sucesso'
    };
  } catch (err: any) {
    return { success: false, message: err.message };
  }
}

export async function confirmPayment(credentials: ISPCredentials, params: {

      success: response.type === 'success',
```

```typescript
import { db } from '../db';
import { isp_configs, isp_unlock_logs } from '../db/schema';
import { eq, and, gte, count } from 'drizzle-orm';
import { getProviderFromConfig } from './isp-provider-factory';
import { sendMessage } from './channel-router';

const DEFAULT_TRUST_UNLOCK_MESSAGE = `✅ Seu sinal foi reativado, {{nome}}!

Seu acesso à internet foi liberado. Lembre-se de realizar o pagamento da fatura em aberto para evitar nova suspensão.

Fatura: *{{valor}}* vencida em *{{vencimento}}*
PIX/Boleto: {{link_boleto}}

Qualquer dúvida, estamos aqui! 😊`;

const DEFAULT_OPERATOR_MESSAGE = `⚠️ *Desbloqueio de Confiança*

Cliente: *{{nome}}*
Telefone: {{telefone}}
Fatura em atraso: *{{valor}}* ({{dias_atraso}} dias)
Vencimento: {{vencimento}}

Desbloqueio concedido automaticamente via WhatsApp.`;

export interface UnlockParams {
  workspaceId: string;
  customerId: string;
  contractId: string;
  customerName: string;
  customerPhone: string;
  invoiceId?: string;
  invoiceValue?: number;
  invoiceDueDate?: string;
  invoiceDaysOverdue?: number;
  invoiceLink?: string;
  unlockType: 'trust' | 'payment_confirmed';
}

export interface UnlockResult {
  success: boolean;
  message: string;
  alreadyUnlocked?: boolean;
  restrictedReason?: string;
}

export async function executeTrustUnlock(params: UnlockParams): Promise<UnlockResult> {
  const { workspaceId, customerId, contractId } = params;

  // 1. Load ISP config
  const config = await db.query.isp_configs.findFirst({
    where: and(
      eq(isp_configs.workspace_id, workspaceId),
      eq(isp_configs.is_active, true)
    )
  });
  if (!config) return { success: false, message: 'ISP não configurado' };

  // 2. Check if trust unlock is enabled
  if (!config.trust_unlock_enabled) {
    return {
      success: false,
      restrictedReason: 'disabled',
      message: 'Desbloqueio de confiança não está disponível no momento.'
    };
  }

  // 3. Check max days overdue restriction
  if (config.trust_unlock_max_days_overdue && params.invoiceDaysOverdue) {
    if (params.invoiceDaysOverdue > config.trust_unlock_max_days_overdue) {
      return {
        success: false,
        restrictedReason: 'max_days_exceeded',
        message: `Não é possível realizar o desbloqueio de confiança com mais de ${config.trust_unlock_max_days_overdue} dias de atraso. Entre em contato com nosso suporte.`
      };
    }
  }

  // 4. Check cooldown — same customer unlock within configured hours
  const cooldownHours = config.trust_unlock_cooldown_hours || 24;
  const cooldownDate = new Date(Date.now() - cooldownHours * 60 * 60 * 1000);

  const recentUnlock = await db.query.isp_unlock_logs.findFirst({
    where: and(
      eq(isp_unlock_logs.workspace_id, workspaceId),
      eq(isp_unlock_logs.sgp_customer_id, customerId),
      eq(isp_unlock_logs.success, true),
      gte(isp_unlock_logs.unlocked_at, cooldownDate)
    )
  });

  if (recentUnlock) {
    return {
      success: false,
      alreadyUnlocked: true,
      message: `Seu sinal já foi desbloqueado recentemente. O desbloqueio de confiança pode ser usado uma vez a cada ${cooldownHours} horas.`
    };
  }

  // 5. Check monthly limit per customer
  if (config.trust_unlock_max_per_month) {
    const startOfMonth = new Date();
    startOfMonth.setDate(1);
    startOfMonth.setHours(0, 0, 0, 0);

    const [monthlyCount] = await db.select({ value: count() })
      .from(isp_unlock_logs)
      .where(and(
        eq(isp_unlock_logs.workspace_id, workspaceId),
        eq(isp_unlock_logs.sgp_customer_id, customerId),
        eq(isp_unlock_logs.success, true),
        gte(isp_unlock_logs.unlocked_at, startOfMonth)
      ));

    if ((monthlyCount?.value || 0) >= config.trust_unlock_max_per_month) {
      return {
        success: false,
        restrictedReason: 'monthly_limit',
        message: `Você atingiu o limite de desbloqueios de confiança deste mês. Entre em contato com o suporte para regularizar sua situação.`
      };
    }
  }

  // 6. Execute unlock in ERP
  const { provider, credentials } = getProviderFromConfig(config);

  const unlockResult = await provider.trustUnlock(credentials, { customerId, contractId });
  const reactivateResult = await provider.reactivateContract(credentials, { customerId, contractId });

  const overallSuccess = unlockResult.success || reactivateResult.success;

  // 7. Log the unlock attempt
  await db.insert(isp_unlock_logs).values({
    workspace_id: workspaceId,
    sgp_customer_id: customerId,
    sgp_contract_id: contractId,
    customer_phone: params.customerPhone,
    customer_name: params.customerName,
    unlock_type: params.unlockType,
    days_overdue_at_unlock: params.invoiceDaysOverdue || 0,
    erp_response: { unlock: unlockResult, reactivate: reactivateResult },
    success: overallSuccess,
    error_message: overallSuccess ? null : `${unlockResult.message} | ${reactivateResult.message}`
  }).catch(e => console.error('[ISP Unlock] Log error:', e));

  if (!overallSuccess) {
    return {
      success: false,
      message: 'Não foi possível realizar o desbloqueio automático. Entre em contato com o suporte.'
    };
  }

  // 8. Send confirmation message to client
  const clientMessage = buildMessage(
    config.trust_unlock_message || DEFAULT_TRUST_UNLOCK_MESSAGE,
    {
      nome: params.customerName?.split(' ')[0] || 'cliente',
      valor: params.invoiceValue ? `R$ ${params.invoiceValue.toFixed(2)}` : '',
      vencimento: params.invoiceDueDate ? new Date(params.invoiceDueDate).toLocaleDateString('pt-BR') : '',
      link_boleto: params.invoiceLink || ''
    }
  );

  await sendMessage({
    workspaceId,
    to: params.customerPhone,
    type: 'text',
    content: clientMessage
  }).catch(e => console.error('[ISP Unlock] Client notification error:', e));

  // 9. Notify operator if configured
  if (config.trust_unlock_operator_notify && config.operator_phone) {
    const operatorMessage = buildMessage(
      DEFAULT_OPERATOR_MESSAGE,
      {
        nome: params.customerName,
        telefone: params.customerPhone,
        valor: params.invoiceValue ? `R$ ${params.invoiceValue.toFixed(2)}` : 'N/D',
        dias_atraso: String(params.invoiceDaysOverdue || 0),
        vencimento: params.invoiceDueDate ? new Date(params.invoiceDueDate).toLocaleDateString('pt-BR') : 'N/D'
      }
    );

    await sendMessage({
      workspaceId,
      to: config.operator_phone,
      type: 'text',
      content: operatorMessage
    }).catch(e => console.error('[ISP Unlock] Operator notification error:', e));
  }

  return {
    success: true,
    message: `Sinal reativado com sucesso!`
  };
}

export async function executePaymentConfirmation(params: UnlockParams): Promise<UnlockResult> {
  const { workspaceId, customerId, contractId } = params;

  const config = await db.query.isp_configs.findFirst({
    where: and(eq(isp_configs.workspace_id, workspaceId), eq(isp_configs.is_active, true))
  });
  if (!config) return { success: false, message: 'ISP não configurado' };

  const { provider, credentials } = getProviderFromConfig(config);

  // 1. Confirm payment in ERP
  const paymentResult = params.invoiceId
    ? await provider.confirmPayment(credentials, {
        invoiceId: params.invoiceId,
        customerId,
        contractId
      })
    : { success: false, message: 'ID da fatura não informado' };

  // 2. Reactivate contract
  const reactivateResult = await provider.reactivateContract(credentials, { customerId, contractId });

  // 3. Trust unlock (unblock access)
  const unlockResult = await provider.trustUnlock(credentials, { customerId, contractId });

  const overallSuccess = paymentResult.success || (reactivateResult.success && unlockResult.success);

  // 4. Log
  await db.insert(isp_unlock_logs).values({
    workspace_id: workspaceId,
    sgp_customer_id: customerId,
    sgp_contract_id: contractId,
    customer_phone: params.customerPhone,
    customer_name: params.customerName,
    unlock_type: 'payment_confirmed',
    days_overdue_at_unlock: params.invoiceDaysOverdue || 0,
    erp_response: { payment: paymentResult, reactivate: reactivateResult, unlock: unlockResult },
    success: overallSuccess,
    error_message: overallSuccess ? null : paymentResult.message
  }).catch(() => {});

  // 5. Notify operator
  if (config.trust_unlock_operator_notify && config.operator_phone) {
    const msg = `✅ *Pagamento Confirmado*\n\nCliente: *${params.customerName}*\nFatura: *${params.invoiceValue ? `R$ ${params.invoiceValue.toFixed(2)}` : 'N/D'}*\nContrato reativado: ${reactivateResult.success ? '✅' : '❌'}\nAcesso liberado: ${unlockResult.success ? '✅' : '❌'}`;
    await sendMessage({ workspaceId, to: config.operator_phone, type: 'text', content: msg }).catch(() => {});
  }

  return { success: overallSuccess, message: paymentResult.message };
}

function buildMessage(template: string, vars: Record<string, string>): string {
  return template.replace(/\{\{(\w+)\}\}/g, (_, k) => vars[k] ?? '');
}
```

```typescript
case 'isp_unlock': {
  const workspaceId = context.workspaceId;
  const action = node.config.action;
  const resolve = (val: string) => val?.replace(/\{\{(\w+)\}\}/g, (_, k) => context.variables[k] ?? '');

          invoiceDueDate: resolve('{{isp_invoice_vencimento_raw}}'),
          invoiceDaysOverdue: parseInt(String(context.variables.isp_invoice_dias_atraso || '0')) || 0,
          invoiceLink: resolve('{{isp_link_boleto}}'),
          unlockType: 'trust'
        });

        context.variables.isp_unlock_success = result.success;
        context.variables.isp_unlock_message = result.message;
        context.variables.isp_unlock_restricted = !!result.restrictedReason;
        context.variables.isp_unlock_restriction_reason = result.restrictedReason || '';
        context.variables.isp_already_unlocked = !!result.alreadyUnlocked;
        break;
      }

      case 'confirm_payment': {
        const result = await executePaymentConfirmation({
          workspaceId,
          customerId: resolve(node.config.customer_id_variable || '{{isp_customer_id}}'),
          contractId: resolve(node.config.contract_id_variable || '{{isp_contract_id}}'),
          customerName: resolve('{{isp_customer_nome}}'),
          customerPhone: context.variables.customerPhone || '',
          invoiceId: resolve(node.config.invoice_id_variable || '{{isp_invoice_id}}'),
          invoiceValue: parseFloat(String(context.variables.isp_invoice_valor_raw || '0')) || undefined,
          unlockType: 'payment_confirmed'
        });

        context.variables.isp_payment_success = result.success;
        context.variables.isp_payment_message = result.message;
        break;
      }
    }

    context.variables.isp_success = true;
  } catch (err: any) {
    context.variables.isp_success = false;
    context.variables.isp_error = err.message;
    console.error('[ISP Unlock Node] Error:', err.message);
  }
  break;
}
```
