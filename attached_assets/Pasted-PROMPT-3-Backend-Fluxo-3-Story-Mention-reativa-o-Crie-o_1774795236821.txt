PROMPT 3 — Backend: Fluxo 3 (Story Mention reativação)
Crie o arquivo `server/services/instagramStoryHandler.ts`:
```ts
import { db } from '../db';
import { leads, instagramConnections } from '../../shared/schema';
import { eq, and } from 'drizzle-orm';
import { sendInstagramDM, upsertInstagramLead } from './instagramService';

export async function processStoryMention(params: {
  workspaceId: string;
  connectionId: string;
  accessToken: string;
  igAccountUserId: string;
  fromIgUserId: string;
  fromIgUsername: string;
  storyId: string;
}) {
  const { workspaceId, accessToken, igAccountUserId,
    fromIgUserId, fromIgUsername } = params;

  // Buscar ou criar lead
  const lead = await upsertInstagramLead(workspaceId, fromIgUserId, fromIgUsername);

  // Verificar se é ex-cliente (status cancelado no CRM)
  const existingLead = await db
    .select()
    .from(leads as any)
    .where(and(
      eq((leads as any).workspaceId, workspaceId),
      eq((leads as any).instagramId, fromIgUserId)
    ))
    .limit(1);

  const isCancelledClient = existingLead[0]?.status === 'cancelado'
    || existingLead[0]?.status === 'churned';

  if (isCancelledClient) {
    // Fluxo 3 — Reativação de ex-cliente
    await sendInstagramDM(accessToken, igAccountUserId, fromIgUserId,
      `Oi ${fromIgUsername}! Que saudade! 😄\n\n` +
      `Vimos que você marcou a gente nos Stories e ficamos felizes!\n\n` +
      `Temos planos novos bem melhores desde a última vez que você era nosso cliente. ` +
      `Quer que eu te mostre as novidades? 🚀`
    );

    // Atualizar status para reengajando
    await db.update(leads as any)
      .set({
        reengajando: true,
        source: 'instagram_story',
        status: 'reengajando',
      })
      .where(eq((leads as any).id, lead.id));

  } else {
    // Novo visitante — DM de boas-vindas
    await sendInstagramDM(accessToken, igAccountUserId, fromIgUserId,
      `Oi ${fromIgUsername}! Obrigado por marcar a gente nos Stories! 🙌\n\n` +
      `Posso te ajudar com alguma coisa? ` +
      `Se quiser saber sobre nossos planos de internet fibra, é só falar! 😊`
    );

    await db.update(leads as any)
      .set({ source: 'instagram_story' })
      .where(eq((leads as any).id, lead.id));
  }

  console.log(`[Story Mention] ${fromIgUsername} — ex-cliente: ${isCancelledClient}`);
}
```

PROMPT 4 — Backend: cron job pós-instalação + lista de espera
No arquivo de jobs agendados do projeto (onde ficam os node-cron jobs — provavelmente `server/jobs/` ou `server/schedulers/`), adicione dois novos jobs:

**Job 1 — Processar disparos pós-instalação Instagram:**
```ts
import cron from 'node-cron';
import { db } from '../db';
import { disparosProgramados } from '../../shared/schema';
import { eq, and, lte } from 'drizzle-orm';
import { sendInstagramDM } from '../services/instagramService';

// Roda a cada 5 minutos
cron.schedule('*/5 * * * *', async () => {
  const now = new Date();

  const pendingDisparos = await db
    .select()
    .from(disparosProgramados as any)
    .where(and(
      eq((disparosProgramados as any).canal, 'instagram'),
      eq((disparosProgramados as any).status, 'pendente'),
      lte((disparosProgramados as any).agendadoPara, now)
    ));

  for (const disparo of pendingDisparos) {
    try {
      const meta = disparo.metadata as any;
      if (!meta?.igUserId || !meta?.accessToken || !meta?.igAccountUserId) continue;

      await sendInstagramDM(
        meta.accessToken,
        meta.igAccountUserId,
        meta.igUserId,
        disparo.mensagem
      );

      await db.update(disparosProgramados as any)
        .set({ status: 'enviado', enviadoEm: new Date() })
        .where(eq((disparosProgramados as any).id, disparo.id));

      console.log(`[Pós-instalação IG] Enviado para ${meta.igUserId}`);
    } catch (err) {
      console.error('[Pós-instalação IG] Erro:', err);
      await db.update(disparosProgramados as any)
        .set({ status: 'erro' })
        .where(eq((disparosProgramados as any).id, disparo.id));
    }
  }
});
```

**Job 2 — Notificar lista de espera quando cobertura expandir:**
```ts
import { instagramWaitlist, leads } from '../../shared/schema';

// Roda 1x por dia às 9h
cron.schedule('0 9 * * *', async () => {
  const { checkCoverageISP } = await import('../services/ispInstagramFlows');

  // Buscar todos os leads em lista de espera ainda não notificados
  const waiting = await db
    .select()
    .from(instagramWaitlist)
    .where(eq(instagramWaitlist.notificado, false));

  for (const entry of waiting) {
    try {
      const coverage = await checkCoverageISP(entry.workspaceId, entry.endereco);

      if (coverage.hasCoverage) {
        // Buscar conexão Instagram do workspace
        const { getActiveInstagramConnection } = await import('../services/instagramService');
        const conn = await getActiveInstagramConnection(entry.workspaceId);
        if (!conn || !entry.igUserId) continue;

        await sendInstagramDM(
          conn.accessToken,
          conn.igUserId,
          entry.igUserId!,
          `🎉 ${entry.igUsername ? `Oi ${entry.igUsername}! ` : ''}Novidade!\n\n` +
          `Chegamos no seu bairro! Agora temos cobertura no seu endereço.\n` +
          `Quer ver os planos disponíveis? Me chama aqui! 🚀`
        );

        await db.update(instagramWaitlist)
          .set({ notificado: true })
          .where(eq(instagramWaitlist.id, entry.id));

        await db.update(leads as any)
          .set({ coberturaStatus: 'disponivel', status: 'reengajando' })
          .where(eq((leads as any).id, entry.leadId));

        console.log(`[Lista de Espera] Notificado: ${entry.igUsername} — ${entry.endereco}`);
      }
    } catch (err) {
      console.error('[Lista de Espera] Erro:', err);
    }
  }
});
```
