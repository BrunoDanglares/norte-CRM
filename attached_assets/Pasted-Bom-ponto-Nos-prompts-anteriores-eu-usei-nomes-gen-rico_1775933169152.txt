### Arquivo 2: `server/services/kanbanArchivalService.ts`

Cria esse arquivo completo:

```typescript
import { db } from "../db";
import { leads } from "../../shared/schema";
import { eq, and, isNull, inArray } from "drizzle-orm";

/**
 * Stages finais por pipeline — exatamente como estão cadastrados no banco.
 *
 * COMERCIAL:
 *   "Instalado / Ativado" → ativado
 *   "Perdido"             → perdido
 *
 * SUPORTE TÉCNICO (3 etapas finais):
 *   "Resolvido"           → ativado
 *   "Visita Técnica"      → ativado
 *   "Escalado / NOC"      → perdido
 *
 * FINANCEIRO (3 etapas finais):
 *   "Pago / Regularizado" → ativado
 *   "Inadimplente"        → perdido
 *   "Cancelado por Débito"→ perdido
 */
const FINAL_STAGE_MAP: Record<string, "ativado" | "perdido"> = {
  // Comercial
  "Instalado / Ativado":  "ativado",
  "Perdido":              "perdido",
  // Suporte Técnico
  "Resolvido":            "ativado",
  "Visita Técnica":       "ativado",
  "Escalado / NOC":       "perdido",
  // Financeiro
  "Pago / Regularizado":  "ativado",
  "Inadimplente":         "perdido",
  "Cancelado por Débito": "perdido",
};

const FINAL_STAGE_NAMES = Object.keys(FINAL_STAGE_MAP);

/**
 * Arquiva um lead imediatamente ao ser movido para etapa final.
 * Chamado pelo frontend/rota quando o lead é arrastado para a coluna final.
 */
export async function archiveLead(
  leadId: number,
  reason: "ativado" | "perdido"
): Promise<void> {
  await db
    .update(leads)
    .set({
      archivedAt: new Date(),
      archivalReason: reason,
    })
    .where(eq(leads.id, leadId));
}

/**
 * Detecta automaticamente o motivo pelo nome do stage.
 * Útil quando o arquivamento vem de uma movimentação de stage sem reason explícito.
 */
export function resolveArchivalReason(stageName: string): "ativado" | "perdido" {
  return FINAL_STAGE_MAP[stageName] ?? "perdido";
}

/**
 * Job de limpeza de fim de expediente.
 * Arquiva todos os leads nas etapas finais que ainda não foram arquivados.
 * Deve ser chamado às 18h (seg-sex) e 12h (sáb).
 */
export async function archiveEndOfShift(): Promise<{ archived: number }> {
  const leadsToArchive = await db
    .select({ id: leads.id, stage: leads.stage })
    .from(leads)
    .where(
      and(
        isNull(leads.archivedAt),
        inArray(leads.stage, FINAL_STAGE_NAMES)
      )
    );

  if (leadsToArchive.length === 0) return { archived: 0 };

  for (const lead of leadsToArchive) {
    const reason = resolveArchivalReason(lead.stage ?? "");
    await db
      .update(leads)
      .set({ archivedAt: new Date(), archivalReason: reason })
      .where(eq(leads.id, lead.id));
  }

  console.log(`[KanbanArchival] ${leadsToArchive.length} leads arquivados (fim de expediente)`);
  return { archived: leadsToArchive.length };
}
```

### 1. Criar `server/routes/history.ts`

Cria esse arquivo completo:

```typescript
import { Router } from "express";
import { db } from "../db";
import {
  conversations,
  protocols,
  conversationSituationTags,
  leads,
} from "../../shared/schema";
import { eq, desc, and, sql, isNull } from "drizzle-orm";
import { requireAuth } from "../middleware/auth";
import { resolveWorkspaceId } from "../utils/workspace";

const router = Router();

// GET /api/history
router.get("/", requireAuth, async (req, res) => {
  try {
    const workspaceId = resolveWorkspaceId(req);
    const { limit = "50", offset = "0" } = req.query;

    const rows = await db
      .select({
        conversationId:    conversations.id,
        contactName:       conversations.contactName,
        contactPhone:      conversations.contactPhone,
        pipeline:          conversations.pipeline,
        status:            conversations.status,
        resolvedAt:        conversations.updatedAt,
        agentName:         conversations.assignedAgent,
        protocolId:        protocols.id,
        protocolNumber:    protocols.protocolNumber,
        leadStage:         leads.stage,
        leadArchivedAt:    leads.archivedAt,
        leadArchivalReason: leads.archivalReason,
      })
      .from(conversations)
      .leftJoin(protocols, eq(protocols.conversationId, conversations.id))
      .leftJoin(leads, eq(leads.conversationId, conversations.id))
      .where(
        and(
          eq(conversations.workspaceId, workspaceId),
          eq(conversations.status, "resolved")
        )
      )
      .orderBy(desc(conversations.updatedAt))
      .limit(parseInt(limit as string))
      .offset(parseInt(offset as string));

    const conversationIds = rows.map((r) => r.conversationId);
    const tagsMap: Record<number, any[]> = {};

    if (conversationIds.length > 0) {
      const allTags = await db
        .select()
        .from(conversationSituationTags)
        .where(
          sql`${conversationSituationTags.conversationId} = ANY(${sql.raw(
            `ARRAY[${conversationIds.join(",")}]`
          )})`
        );

      for (const tag of allTags) {
        if (!tagsMap[tag.conversationId]) tagsMap[tag.conversationId] = [];
        tagsMap[tag.conversationId].push(tag);
      }
    }

    const result = rows.map((row) => ({
      ...row,
      tags: tagsMap[row.conversationId] || [],
    }));

    res.json({ data: result, total: result.length });
  } catch (error) {
    console.error("[History] Erro:", error);
    res.status(500).json({ error: "Erro interno" });
  }
});

// GET /api/history/metrics
router.get("/metrics", requireAuth, async (req, res) => {
  try {
    const workspaceId = resolveWorkspaceId(req);

    const [totalResolved] = await db
      .select({ count: sql<number>`count(*)` })
      .from(conversations)
      .where(
        and(
          eq(conversations.workspaceId, workspaceId),
          eq(conversations.status, "resolved")
        )
      );

    const topTags = await db
      .select({
        tagSlug: conversationSituationTags.tagSlug,
        count:   sql<number>`count(*)`,
      })
      .from(conversationSituationTags)
      .where(eq(conversationSituationTags.workspaceId, workspaceId))
      .groupBy(conversationSituationTags.tagSlug)
      .orderBy(desc(sql`count(*)`))
      .limit(10);

    const [ativados] = await db
      .select({ count: sql<number>`count(*)` })
      .from(leads)
      .where(
        and(
          eq(leads.workspaceId, workspaceId),
          eq(leads.archivalReason, "ativado")
        )
      );

    const [perdidos] = await db
      .select({ count: sql<number>`count(*)` })
      .from(leads)
      .where(
        and(
          eq(leads.workspaceId, workspaceId),
          eq(leads.archivalReason, "perdido")
        )
      );

    // Breakdown por pipeline
    const byPipeline = await db
      .select({
        pipeline: leads.pipeline,
        reason:   leads.archivalReason,
        count:    sql<number>`count(*)`,
      })
      .from(leads)
      .where(
        and(
          eq(leads.workspaceId, workspaceId),
          sql`${leads.archivedAt} IS NOT NULL`
        )
      )
      .groupBy(leads.pipeline, leads.archivalReason);

    res.json({
      totalResolved: Number(totalResolved?.count || 0),
      ativados:      Number(ativados?.count || 0),
      perdidos:      Number(perdidos?.count || 0),
      topTags,
      byPipeline,
    });
  } catch (error) {
    console.error("[History] Métricas erro:", error);
    res.status(500).json({ error: "Erro interno" });
  }
});

// POST /api/history/conversations/:id/tags — tag manual
router.post("/conversations/:id/tags", requireAuth, async (req, res) => {
  try {
    const workspaceId   = resolveWorkspaceId(req);
    const conversationId = parseInt(req.params.id);
    const { situationCode } = req.body;
    const userId = (req as any).user?.id;

    const { applyManualTag } = await import("../services/situationTagService");
    await applyManualTag(workspaceId, conversationId, situationCode, userId);

    res.json({ success: true });
  } catch (error: any) {
    res.status(400).json({ error: error.message });
  }
});

// DELETE /api/history/conversations/:id/tags/:tagId
router.delete("/conversations/:id/tags/:tagId", requireAuth, async (req, res) => {
  try {
    const conversationId = parseInt(req.params.id);
    const tagId          = parseInt(req.params.tagId);

    const { removeTag } = await import("../services/situationTagService");
    await removeTag(conversationId, tagId);

    res.json({ success: true });
  } catch (error) {
    res.status(500).json({ error: "Erro interno" });
  }
});

// PATCH /api/history/leads/:id/archive — arquivamento imediato ao mover card
router.patch("/leads/:id/archive", requireAuth, async (req, res) => {
  try {
    const leadId = parseInt(req.params.id);
    const { reason } = req.body as { reason: "ativado" | "perdido" };

    const { archiveLead } = await import("../services/kanbanArchivalService");
    await archiveLead(leadId, reason);

    res.json({ success: true });
  } catch (error) {
    res.status(500).json({ error: "Erro interno" });
  }
});

export default router;
```

```typescript
      if (activeTab === "protocolos") {
        const res  = await apiFetch("/api/protocols");
        const json = await res.json();
        setProtocols(json.protocols || json || []);
      }
    } finally {
      setLoading(false);
    }
  }

  function openConversation(conversationId: number) {
    if (!conversationId) return;
    window.location.href = `/chat?conversationId=${conversationId}`;
  }

  function getPipelineBadgeClass(pipeline: string) {
    if (!pipeline) return "badge-gray";
    const p = pipeline.toLowerCase();
    if (p.includes("financ")) return "badge-financial";
    if (p.includes("suport")) return "badge-support";
    if (p.includes("comerc")) return "badge-commercial";
    return "badge-gray";
  }

  function getTagColor(tagSlug: string) {
    if (!tagSlug) return "";
    if (tagSlug.includes("cancel") || tagSlug.includes("inadim") || tagSlug.includes("escalado"))
      return "tag-red";
    if (tagSlug.includes("contrato") || tagSlug.includes("reativ") || tagSlug.includes("pago") ||
        tagSlug.includes("levantada") || tagSlug.includes("regulariz") || tagSlug.includes("dados-coletados"))
      return "tag-green";
    if (tagSlug.includes("os-aberta") || tagSlug.includes("downgrade") || tagSlug.includes("contestacao") ||
        tagSlug.includes("visita"))
      return "tag-yellow";
    if (tagSlug.includes("onu") || tagSlug.includes("wifi") || tagSlug.includes("lentidao") ||
        tagSlug.includes("instab") || tagSlug.includes("roteador"))
      return "tag-blue";
    return "tag-purple";
  }

  // Label legível do stage final
  function getStageLabel(row: any): string {
    if (row.leadStage) return row.leadStage;
    if (row.leadArchivalReason === "ativado") return "Ativado";
    if (row.leadArchivalReason === "perdido") return "Perdido";
    return "—";
  }

  const TABS = [
    { key: "historico",  label: "Histórico & Controle" },
    { key: "protocolos", label: "Protocolos" },
    { key: "relatorio",  label: "Relatório Kanban" },
    { key: "tags",       label: "Situações & Tags" },
  ];

  return (
    <div className="history-page">
      <div className="history-header">
        <h1>Central de Atendimentos</h1>
        <p>Conversas resolvidas, protocolos e situações ISP em um só lugar</p>
      </div>

      <div className="history-tabs">
        {TABS.map((tab) => (
          <button
            key={tab.key}
            className={`history-tab ${activeTab === tab.key ? "active" : ""}`}
            onClick={() => setActiveTab(tab.key as ActiveTab)}
          >
            {tab.label}
          </button>
        ))}
      </div>

      {/* Histórico */}
      {activeTab === "historico" && (
        <div className="history-content">
          {loading ? <div className="loading-state">Carregando...</div> : (
            <div className="history-table-wrap">
              <table className="history-table">
                <thead>
                  <tr>
                    <th>Protocolo</th><th>Contato</th><th>Setor</th>
                    <th>Etapa final</th><th>Resultado</th><th>Tags</th>
                    <th>Agente</th><th>Data</th><th>Ação</th>
                  </tr>
                </thead>
                <tbody>
                  {historyData.map((row) => (
                    <tr key={row.conversationId}>
                      <td><span className="proto-number">{row.protocolNumber || "—"}</span></td>
                      <td>
                        <div className="contact-name">{row.contactName || "—"}</div>
                        <div className="contact-phone">{row.contactPhone}</div>
                      </td>
                      <td>
                        <span className={`pipeline-badge ${getPipelineBadgeClass(row.pipeline)}`}>
                          {row.pipeline || "—"}
                        </span>
                      </td>
                      <td className="cell-stage">{getStageLabel(row)}</td>
                      <td>
                        {row.leadArchivalReason === "ativado" && (
                          <span className="result-badge result-ativado">✓ Ativado</span>
                        )}
                        {row.leadArchivalReason === "perdido" && (
                          <span className="result-badge result-perdido">✗ Perdido</span>
                        )}
                        {!row.leadArchivalReason && <span className="no-tag">—</span>}
                      </td>
                      <td>
                        <div className="tags-cell">
                          {(row.tags || []).map((tag: any) => (
                            <span key={tag.id} className={`situation-tag ${getTagColor(tag.tagSlug)}`}>
                              {tag.tagSlug}
                            </span>
                          ))}
                          {(!row.tags || row.tags.length === 0) && (
                            <span className="no-tag">sem tag</span>
                          )}
                        </div>
                      </td>
                      <td>{row.agentName || "—"}</td>
                      <td>{row.resolvedAt ? new Date(row.resolvedAt).toLocaleDateString("pt-BR") : "—"}</td>
                      <td>
                        <button
                          className="btn-open-conv"
                          onClick={() => openConversation(row.conversationId)}
                          disabled={!row.conversationId}
                        >
                          ↗ Conversa
                        </button>
                      </td>
                    </tr>
                  ))}
                  {historyData.length === 0 && (
                    <tr><td colSpan={9} className="empty-state">Nenhuma conversa resolvida.</td></tr>
                  )}
                </tbody>
              </table>
            </div>
          )}
        </div>
      )}

      {/* Protocolos */}
      {activeTab === "protocolos" && (
        <div className="history-content">
          {loading ? <div className="loading-state">Carregando...</div> : (
            <div className="history-table-wrap">
              <table className="history-table">
                <thead>
                  <tr>
                    <th>Número</th><th>Contato</th><th>Título</th><th>Categoria</th>
                    <th>Status</th><th>Prioridade</th><th>SLA</th><th>Ação</th>
                  </tr>
                </thead>
                <tbody>
                  {protocols.map((proto) => (
                    <tr key={proto.id}>
                      <td><span className="proto-number">{proto.protocolNumber}</span></td>
                      <td>{proto.contactName || "—"}</td>
                      <td>{proto.title}</td>
                      <td>{proto.category || "Geral"}</td>
                      <td>
                        <span className={`status-badge status-${proto.status}`}>
                          {proto.status}
                        </span>
                      </td>
                      <td>{proto.priority}</td>
                      <td>
                        <div className={`sla-indicator ${proto.slaViolated ? "sla-violated" : "sla-ok"}`} />
                      </td>
                      <td>
                        <button
                          className="btn-open-conv"
                          onClick={() => openConversation(proto.conversationId)}
                          disabled={!proto.conversationId}
                          title={!proto.conversationId ? "Sem conversa vinculada" : "Abrir conversa"}
                        >
                          ↗ Conversa
                        </button>
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          )}
        </div>
      )}

      {/* Relatório */}
      {activeTab === "relatorio" && (
        <div className="history-content">
          {loading || !metrics ? <div className="loading-state">Carregando métricas...</div> : (
            <>
              <div className="metrics-grid">
                <div className="metric-card">
                  <div className="metric-label">Total resolvido</div>
                  <div className="metric-value">{metrics.totalResolved}</div>
                </div>
                <div className="metric-card metric-green">
                  <div className="metric-label">Ativados</div>
                  <div className="metric-value">{metrics.ativados}</div>
                </div>
                <div className="metric-card metric-red">
                  <div className="metric-label">Perdidos / Escalados</div>
                  <div className="metric-value">{metrics.perdidos}</div>
                </div>
                <div className="metric-card">
                  <div className="metric-label">Taxa retenção</div>
                  <div className="metric-value">
                    {metrics.ativados + metrics.perdidos > 0
                      ? Math.round((metrics.ativados / (metrics.ativados + metrics.perdidos)) * 100)
                      : 0}%
                  </div>
                </div>
              </div>

              {/* Breakdown por pipeline */}
              {metrics.byPipeline && metrics.byPipeline.length > 0 && (
                <div className="report-section">
                  <h3>Por setor</h3>
                  <div className="pipeline-breakdown">
                    {["Comercial", "Suporte Técnico", "Financeiro"].map((pipe) => {
                      const rows = metrics.byPipeline.filter((r: any) =>
                        r.pipeline?.toLowerCase().includes(pipe.split(" ")[0].toLowerCase())
                      );
                      const atv = rows.find((r: any) => r.reason === "ativado")?.count || 0;
                      const per = rows.find((r: any) => r.reason === "perdido")?.count || 0;
                      const total = Number(atv) + Number(per);
                      return (
                        <div key={pipe} className="pipeline-row">
                          <span className={`pipeline-badge ${getPipelineBadgeClass(pipe)}`}>{pipe}</span>
                          <span className="pipeline-stat stat-green">✓ {atv}</span>
                          <span className="pipeline-stat stat-red">✗ {per}</span>
                          <span className="pipeline-stat">Total: {total}</span>
                        </div>
                      );
                    })}
                  </div>
                </div>
              )}

              <div className="report-section">
                <h3>Top situações atendidas</h3>
                {(metrics.topTags || []).map((t: any) => (
                  <div key={t.tagSlug} className="bar-row">
                    <span className="bar-label">{t.tagSlug}</span>
                    <div className="bar-track">
                      <div
                        className="bar-fill"
                        style={{
                          width: `${Math.min(100, Math.round(
                            (t.count / (metrics.topTags[0]?.count || 1)) * 100
                          ))}%`,
                        }}
                      />
                    </div>
                    <span className="bar-value">{t.count}</span>
                  </div>
                ))}
              </div>
            </>
          )}
        </div>
      )}

      {/* Situações & Tags */}
      {activeTab === "tags" && (
        <div className="history-content">
          <div className="tags-reference-grid">
            <div className="tag-group">
              <div className="tag-group-title financial">
                💰 Financeiro
                <span className="pipeline-stages-hint">
                  Etapas finais: Pago/Regularizado · Inadimplente · Cancelado por Débito
                </span>
              </div>
              {Object.entries(SITUATION_TAG_MAP).filter(([k]) => k.startsWith("F")).map(([code, slug]) => (
                <div key={code} className="tag-reference-row">
                  <span className="situation-code">{code}</span>
                  <span className="situation-slug tag-green">{slug}</span>
                </div>
              ))}
            </div>
            <div className="tag-group">
              <div className="tag-group-title support">
                🔧 Suporte Técnico
                <span className="pipeline-stages-hint">
                  Etapas finais: Resolvido · Visita Técnica · Escalado/NOC
                </span>
              </div>
              {Object.entries(SITUATION_TAG_MAP).filter(([k]) => k.startsWith("S")).map(([code, slug]) => (
                <div key={code} className="tag-reference-row">
                  <span className="situation-code">{code}</span>
                  <span className="situation-slug tag-blue">{slug}</span>
                </div>
              ))}
            </div>
            <div className="tag-group">
              <div className="tag-group-title commercial">
                🛒 Comercial
                <span className="pipeline-stages-hint">
                  Etapas finais: Instalado/Ativado · Perdido
                </span>
              </div>
              {Object.entries(SITUATION_TAG_MAP).filter(([k]) => k.startsWith("C")).map(([code, slug]) => (
                <div key={code} className="tag-reference-row">
                  <span className="situation-code">{code}</span>
                  <span className="situation-slug tag-purple">{slug}</span>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
```

### 5. Disparar arquivamento no drag-and-drop do kanban

Localiza o handler de movimentação de card no kanban (onde o `stage` do lead é atualizado via drag).

Adiciona após a chamada de update do stage:

```typescript
const FINAL_STAGE_MAP: Record<string, "ativado" | "perdido"> = {
  "Instalado / Ativado":  "ativado",
  "Perdido":              "perdido",
  "Resolvido":            "ativado",
  "Visita Técnica":       "ativado",
  "Escalado / NOC":       "perdido",
  "Pago / Regularizado":  "ativado",
  "Inadimplente":         "perdido",
  "Cancelado por Débito": "perdido",
};

if (FINAL_STAGE_MAP[novoStage]) {
  apiFetch(`/api/history/leads/${leadId}/archive`, {
    method: "PATCH",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ reason: FINAL_STAGE_MAP[novoStage] }),
  }).catch(console.error); // fire-and-forget
}
```
