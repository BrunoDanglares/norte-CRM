```tsimport { Router } from 'express';
import { db } from '../db';
import { instaProspectFlows, instaProspectSessions } from '../../shared/schema';
import { eq, and, desc, sql } from 'drizzle-orm';
import { requireAuth } from '../middleware/auth';
import { resolveWorkspaceId } from '../utils/workspace';const router = Router();// GET /api/insta-prospect/flows — listar fluxos do workspace
router.get('/flows', requireAuth, async (req, res) => {
try {
const workspaceId = resolveWorkspaceId(req);
const flows = await db
.select()
.from(instaProspectFlows)
.where(eq(instaProspectFlows.workspaceId, workspaceId))
.orderBy(desc(instaProspectFlows.createdAt));
res.json(flows);
} catch (err) {
res.status(500).json({ error: 'Erro ao listar fluxos' });
}
});// POST /api/insta-prospect/flows — criar fluxo
router.post('/flows', requireAuth, async (req, res) => {
try {
const workspaceId = resolveWorkspaceId(req);
const {
nome, tipo, keyword, keywordMatchType, postId,
publicReply, firstMessage,
aiPersona, aiSystemPrompt, aiObjective,
finalAction, assignStrategy, autoTags,
} = req.body;if (!nome || !tipo || !aiSystemPrompt) {
  return res.status(400).json({ error: 'nome, tipo e aiSystemPrompt são obrigatórios' });
}const [flow] = await db.insert(instaProspectFlows).values({
  workspaceId, nome, tipo,
  keyword, keywordMatchType, postId,
  publicReply, firstMessage,
  aiPersona, aiSystemPrompt, aiObjective,
  finalAction, assignStrategy,
  autoTags: autoTags || [],
}).returning();res.json(flow);
} catch (err) {
res.status(500).json({ error: 'Erro ao criar fluxo' });
}
});// PATCH /api/insta-prospect/flows/:id — editar fluxo
router.patch('/flows/:id', requireAuth, async (req, res) => {
try {
const workspaceId = resolveWorkspaceId(req);
const [flow] = await db
.update(instaProspectFlows)
.set({ ...req.body, updatedAt: new Date() })
.where(and(
eq(instaProspectFlows.id, req.params.id),
eq(instaProspectFlows.workspaceId, workspaceId)
))
.returning();
res.json(flow);
} catch (err) {
res.status(500).json({ error: 'Erro ao atualizar fluxo' });
}
});// PATCH /api/insta-prospect/flows/:id/toggle — ativar/pausar
router.patch('/flows/:id/toggle', requireAuth, async (req, res) => {
try {
const workspaceId = resolveWorkspaceId(req);
const [current] = await db
.select()
.from(instaProspectFlows)
.where(and(
eq(instaProspectFlows.id, req.params.id),
eq(instaProspectFlows.workspaceId, workspaceId)
))
.limit(1);if (!current) return res.status(404).json({ error: 'Fluxo não encontrado' });const [updated] = await db
  .update(instaProspectFlows)
  .set({ ativo: !current.ativo, updatedAt: new Date() })
  .where(eq(instaProspectFlows.id, req.params.id))
  .returning();res.json(updated);
} catch (err) {
res.status(500).json({ error: 'Erro ao alternar fluxo' });
}
});// DELETE /api/insta-prospect/flows/:id
router.delete('/flows/:id', requireAuth, async (req, res) => {
try {
const workspaceId = resolveWorkspaceId(req);
await db.delete(instaProspectFlows).where(and(
eq(instaProspectFlows.id, req.params.id),
eq(instaProspectFlows.workspaceId, workspaceId)
));
res.json({ ok: true });
} catch (err) {
res.status(500).json({ error: 'Erro ao deletar fluxo' });
}
});// GET /api/insta-prospect/sessions — sessões/conversas ativas
router.get('/sessions', requireAuth, async (req, res) => {
try {
const workspaceId = resolveWorkspaceId(req);
const sessions = await db

});// GET /api/insta-prospect/stats — métricas do módulo
router.get('/stats', requireAuth, async (req, res) => {
try {
const workspaceId = resolveWorkspaceId(req);const [totals] = await db

    transferidos: sql<number>`count(*) filter (where ${instaProspectSessions.status} = 'transferido')`,
  })
  .from(instaProspectSessions)
