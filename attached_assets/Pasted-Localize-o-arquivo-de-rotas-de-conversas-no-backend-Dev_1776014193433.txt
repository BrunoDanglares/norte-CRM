ROTA 1 — Agente atribui equipe
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
router.post('/conversations/:id/assign-team', requireAuth, async (req, res) => {
  try {
    const conversationId = parseInt(req.params.id);
    const workspaceId = resolveWorkspaceId(req);
    const { team_id, team_name, pipeline_etapa } = req.body;

    if (!team_id || !team_name) {
      return res.status(400).json({ error: 'team_id e team_name são obrigatórios' });
    }

    // Verifica se já está atribuída à mesma equipe (evita update desnecessário)
    const current = await pool.query(
      `SELECT assigned_team_id FROM conversations WHERE id = $1 AND workspace_id = $2`,
      [conversationId, workspaceId]
    );
    if (!current.rows.length) return res.status(404).json({ error: 'Conversa não encontrada' });
    if (current.rows[0].assigned_team_id === team_id) return res.json({ success: true, unchanged: true });

    // Atualiza equipe e pipeline
    const updates: string[] = [
      'assigned_team_id = $1',
      'pipeline = $2',
      'updated_at = NOW()'
    ];
    const params: any[] = [team_id, team_name];

    if (pipeline_etapa) {
      updates.push(`pipeline_etapa = $${params.length + 1}`);
      params.push(pipeline_etapa);
    }

    params.push(conversationId, workspaceId);
    await pool.query(
      `UPDATE conversations SET ${updates.join(', ')}
       WHERE id = $${params.length - 1} AND workspace_id = $${params.length}`,
      params
    );

    // Salva mensagem de sistema no histórico
    await pool.query(
      `INSERT INTO messages (conversation_id, workspace_id, direction, tipo, conteudo, status, created_at)
       VALUES ($1, $2, 'system', 'system', $3, 'sent', NOW())`,
      [conversationId, workspaceId, `🔀 Conversa atribuída à equipe ${team_name} pelo agente`]
    );

    // Broadcast WebSocket
    broadcastToWorkspace(workspaceId, 'conversation_updated', {
      conversationId,
      assigned_team_id: team_id,
      team_name,
      pipeline: team_name,
    });

    return res.json({ success: true });
  } catch (error) {
    console.error('[assign-team]', error);
    return res.status(500).json({ error: 'Erro interno' });
  }
});

ROTA 2 — Atendente assume a conversa
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
router.post('/conversations/:id/assume', requireAuth, async (req, res) => {
  try {
    const conversationId = parseInt(req.params.id);
    const workspaceId = resolveWorkspaceId(req);
    const userName = (req as any).user?.nome || (req as any).user?.name || 'Atendente';
    const userId = (req as any).user?.id;

    // Verifica se já tem atendente
    const current = await pool.query(
      `SELECT assigned_user_id, assigned_user_name FROM conversations
       WHERE id = $1 AND workspace_id = $2`,
      [conversationId, workspaceId]
    );
    if (!current.rows.length) return res.status(404).json({ error: 'Conversa não encontrada' });

    const { assigned_user_id, assigned_user_name } = current.rows[0];
    if (assigned_user_id && assigned_user_id !== userId) {
      return res.status(409).json({
        error: `Conversa já está sendo atendida por ${assigned_user_name}`
      });
    }

    await pool.query(
      `UPDATE conversations
       SET assigned_user_id = $1, assigned_user_name = $2, pendente = false, updated_at = NOW()
       WHERE id = $3 AND workspace_id = $4`,
      [userId, userName, conversationId, workspaceId]
    );

    await pool.query(
      `INSERT INTO messages (conversation_id, workspace_id, direction, tipo, conteudo, status, created_at)
       VALUES ($1, $2, 'system', 'system', $3, 'sent', NOW())`,
      [conversationId, workspaceId, `✅ ${userName} assumiu o atendimento`]
    );

    broadcastToWorkspace(workspaceId, 'conversation_updated', {
      conversationId,
      assigned_user_id: userId,
      assigned_user_name: userName,
      pendente: false,
    });

    return res.json({ success: true, agente: userName });
  } catch (error) {
    console.error('[assume]', error);
    return res.status(500).json({ error: 'Erro interno' });
  }
});

ROTA 3 — Transferir para outra equipe
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
router.post('/conversations/:id/transfer-team', requireAuth, async (req, res) => {
  try {
    const conversationId = parseInt(req.params.id);
    const workspaceId = resolveWorkspaceId(req);
    const { team_id, team_name, motivo } = req.body;

    if (!team_id || !team_name) {
      return res.status(400).json({ error: 'team_id e team_name são obrigatórios' });
    }

    // Busca o primeiro estágio da nova equipe/pipeline
    const stageResult = await pool.query(
      `SELECT nome FROM pipeline_stages
       WHERE workspace_id = $1 AND pipeline_name = $2
       ORDER BY ordem ASC LIMIT 1`,
      [workspaceId, team_name]
    );
    const firstStage = stageResult.rows[0]?.nome || null;

    await pool.query(
      `UPDATE conversations
       SET assigned_team_id = $1,
           pipeline = $2,
           pipeline_etapa = $3,
           assigned_user_id = NULL,
           assigned_user_name = NULL,
           pendente = true,
           updated_at = NOW()
       WHERE id = $4 AND workspace_id = $5`,
      [team_id, team_name, firstStage, conversationId, workspaceId]
    );

    const msg = motivo
      ? `🔀 Transferido para equipe ${team_name}. Motivo: ${motivo}`
      : `🔀 Transferido para equipe ${team_name}`;

    await pool.query(
      `INSERT INTO messages (conversation_id, workspace_id, direction, tipo, conteudo, status, created_at)
       VALUES ($1, $2, 'system', 'system', $3, 'sent', NOW())`,
      [conversationId, workspaceId, msg]
    );

    broadcastToWorkspace(workspaceId, 'conversation_updated', {
      conversationId,
      assigned_team_id: team_id,
      team_name,
      pipeline: team_name,
      pipeline_etapa: firstStage,
      pendente: true,
      assigned_user_id: null,
      assigned_user_name: null,
    });

    return res.json({ success: true });
  } catch (error) {
    console.error('[transfer-team]', error);
    return res.status(500).json({ error: 'Erro interno' });
  }
});
