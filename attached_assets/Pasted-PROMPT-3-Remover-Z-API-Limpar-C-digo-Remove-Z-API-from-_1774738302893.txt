```typescript
export async function sendMessage(params: SendMessageParams): Promise<SendMessageResult> {
  const { workspaceId } = params;

  // PRIORITY 1: Meta Cloud API
  const metaConnection = await db.query.whatsapp_official_connections.findFirst({
    where: and(
      eq(whatsapp_official_connections.workspace_id, workspaceId),
      eq(whatsapp_official_connections.status, 'active')
    )
  });
  if (metaConnection) return sendViaMeta(params, metaConnection);

  // PRIORITY 2: Web.js
  const wwebSession = getActiveSessionForWorkspace(workspaceId);
  if (wwebSession) return sendViaWebJS(params, workspaceId);

  // No channel available
  console.error('[ChannelRouter] No active WhatsApp channel for workspace:', workspaceId);
  return { 
    success: false, 
    error: 'Nenhum canal WhatsApp configurado. Conecte via WhatsApp Web.js ou API Oficial.',
    channel: 'none' as any
  };
}
```
