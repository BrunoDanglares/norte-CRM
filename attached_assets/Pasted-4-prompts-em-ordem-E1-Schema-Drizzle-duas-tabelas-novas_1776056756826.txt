Prompt E2 — ispMetricsService + instrumentação
No backend ChatBanana (Node/Express/TypeScript), precisa criar o serviço de métricas ISP e instrumentar os arquivos existentes para emitir eventos. O serviço deve ser sempre fire-and-forget — nunca bloqueia nem propaga erro para o agente.

**1. Cria `server/services/ispMetricsService.ts`:**

```typescript
import { db } from '../db';
import { ispSessionMetrics, ispSystemEvents } from '../db/schema';
import { eq } from 'drizzle-orm';

// Fire-and-forget: chama sem await, nunca lança
function safe(fn: () => Promise<void>) {
  fn().catch(err => console.error('[Metrics] erro silencioso:', err));
}

export const ispMetrics = {

  // Chamado no início de cada sessão (primeiro turn)
  sessionStart(params: {
    workspaceId: string;
    conversationId: string;
    phone: string;
  }) {
    safe(async () => {
      await db.insert(ispSessionMetrics).values({
        ...params,
        startedAt: new Date(),
      }).onConflictDoNothing(); // idempotente — ignora se já existe
    });
  },

  // Atualiza campos da sessão (pode chamar múltiplas vezes)
  sessionUpdate(conversationId: string, patch: Partial<typeof ispSessionMetrics.$inferInsert>) {
    safe(async () => {
      await db.update(ispSessionMetrics)
        .set(patch)
        .where(eq(ispSessionMetrics.conversationId, conversationId));
    });
  },

  // Chamado quando a sessão termina (resolução, escalação, max turns)
  sessionEnd(conversationId: string, resolutionType: string, latencyMs?: number) {
    safe(async () => {
      await db.update(ispSessionMetrics)
        .set({
          resolutionType,
          resolvedAt: new Date(),
          ...(latencyMs ? { totalLatencyMs: latencyMs } : {}),
        })
        .where(eq(ispSessionMetrics.conversationId, conversationId));
    });
  },

  // Evento de infraestrutura — circuit breaker, timeout, phone lock, etc.
  systemEvent(params: {
    workspaceId?: string;
    eventType: string;
    label?: string;
    metadata?: Record<string, unknown>;
  }) {
    safe(async () => {
      await db.insert(ispSystemEvents).values({
        ...params,
        createdAt: new Date(),
      });
    });
  },
};
```

Prompt E3 — Rotas API
No backend ChatBanana (Node/Express/TypeScript/Drizzle), precisa de 3 endpoints para o dashboard de métricas ISP. Cria o arquivo `server/routes/isp-metrics.ts`:

```typescript
import { Router } from 'express';
import { db } from '../db';
import { ispSessionMetrics, ispSystemEvents } from '../db/schema';
import { eq, gte, desc, sql, and } from 'drizzle-orm';
import { requireAuth } from '../middleware/auth'; // adapta para o middleware real

const router = Router();

// GET /api/isp/metrics/summary?workspaceId=xxx&since=2024-01-01
// Cards do dashboard: totais, taxas, latências
router.get('/summary', requireAuth, async (req, res) => {
  try {
    const { workspaceId, since } = req.query as Record<string, string>;
    const sinceDate = since ? new Date(since) : new Date(Date.now() - 7 * 24 * 60 * 60 * 1000);

    const [sessions] = await db
      .select({
        total:       sql<number>`count(*)`,
        resolved:    sql<number>`count(*) filter (where resolution_type = 'resolved')`,
        escalated:   sql<number>`count(*) filter (where resolution_type = 'escalated')`,
        maxTurns:    sql<number>`count(*) filter (where resolution_type = 'max_turns')`,
        avgTurns:    sql<number>`round(avg(total_turns), 1)`,
        avgLatencyMs:sql<number>`round(avg(total_latency_ms), 0)`,
        p95LatencyMs:sql<number>`percentile_disc(0.95) within group (order by total_latency_ms)`,
        erpDegraded: sql<number>`count(*) filter (where erp_unavailable = true)`,
      })
      .from(ispSessionMetrics)
      .where(and(
        eq(ispSessionMetrics.workspaceId, workspaceId),
        gte(ispSessionMetrics.startedAt, sinceDate)
      ));

    const [events] = await db
      .select({
        circuitBreakerOpens: sql<number>`count(*) filter (where event_type = 'CIRCUIT_BREAKER_OPEN')`,
        timeoutsErp:         sql<number>`count(*) filter (where event_type = 'TIMEOUT_ERP')`,
        timeoutsGpt:         sql<number>`count(*) filter (where event_type = 'TIMEOUT_GPT')`,
        phoneLockDiscards:   sql<number>`count(*) filter (where event_type = 'PHONE_LOCK_DISCARD')`,
      })
      .from(ispSystemEvents)
      .where(and(
        eq(ispSystemEvents.workspaceId, workspaceId),
        gte(ispSystemEvents.createdAt, sinceDate)
      ));

    res.json({ sessions, events });
  } catch (err) {
    console.error('[ISP Metrics] summary error:', err);
    res.status(500).json({ error: 'Erro ao buscar métricas' });
  }
});

// GET /api/isp/metrics/events?workspaceId=xxx&limit=50
// Log de eventos do sistema (circuit breaker, timeouts, etc.)
router.get('/events', requireAuth, async (req, res) => {
  try {
    const { workspaceId, limit = '50' } = req.query as Record<string, string>;

    const events = await db
      .select()
      .from(ispSystemEvents)
      .where(eq(ispSystemEvents.workspaceId, workspaceId))
      .orderBy(desc(ispSystemEvents.createdAt))
      .limit(Number(limit));

    res.json({ events });
  } catch (err) {
    console.error('[ISP Metrics] events error:', err);
    res.status(500).json({ error: 'Erro ao buscar eventos' });
  }
});

// GET /api/isp/metrics/sessions?workspaceId=xxx&limit=100
// Lista de sessões recentes com detalhes
router.get('/sessions', requireAuth, async (req, res) => {
  try {
    const { workspaceId, limit = '100' } = req.query as Record<string, string>;

    const sessions = await db
      .select()
      .from(ispSessionMetrics)
      .where(eq(ispSessionMetrics.workspaceId, workspaceId))
      .orderBy(desc(ispSessionMetrics.startedAt))
      .limit(Number(limit));

    res.json({ sessions });
  } catch (err) {
    console.error('[ISP Metrics] sessions error:', err);
    res.status(500).json({ error: 'Erro ao buscar sessões' });
  }
});

export default router;
```

Registra o router no servidor principal (onde os outros routers ISP já estão registrados):
```typescript

```

- Taxa de escalação (escalated + max_turns / total em %)

**Log de eventos do sistema** — lista reversa cronológica com badge colorido por tipo:
- CIRCUIT_BREAKER_OPEN → vermelho
- TIMEOUT_ERP / TIMEOUT_GPT → amarelo
