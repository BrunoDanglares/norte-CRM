async function callAI(systemPrompt: string, history: any[], newMessage: string): Promise<string> {
const messages = [
{ role: 'system', content: systemPrompt },
...history.map((h: any) => ({ role: h.role, content: h.content })),
{ role: 'user', content: newMessage },
];const res = await fetch('https://api.openai.com/v1/chat/completions', {
method: 'POST',
headers: {
'Content-Type': 'application/json',
Authorization: Bearer ${OPENAI_API_KEY},
},
body: JSON.stringify({
model: 'gpt-4o-mini',
messages,
max_tokens: 300,
temperature: 0.7,
}),
});const data = await res.json();
return data.choices?.[0]?.message?.content?.trim() || 'Desculpe, tente novamente em instantes.';
}// ─── Processar DM recebida (roteador principal) ──────────────────────────────

.select()

eq(instaProspectFlows.tipo, 'dm_received')

const flow = params.flow || await db.query.instaProspectFlows?.findFirst({
where: (t, { eq }) => eq(t.id, session.flowId),
});if (!flow) return;const history = (session.conversationHistory as any[]) || [];// Chamar IA
const aiResponse = await callAI(flow.aiSystemPrompt, history, newMessage);// Enviar resposta ao usuário
await sendInstagramDM(accessToken, igAccountUserId, session.igUserId, aiResponse);// Atualizar histórico na sessão
const updatedHistory = [
...history,
{ role: 'user', content: newMessage, timestamp: new Date().toISOString() },
{ role: 'assistant', content: aiResponse, timestamp: new Date().toISOString() },
];// Verificar se a IA sinalizou qualificação concluída

}
