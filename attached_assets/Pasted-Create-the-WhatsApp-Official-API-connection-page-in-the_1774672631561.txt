### Section 2: Embedded Signup Implementation

The Embedded Signup opens a Meta popup. Implement this exactly:
```typescript
// Load Facebook SDK on component mount
useEffect(() => {
  if (window.FB) return; // already loaded
  
  window.fbAsyncInit = function() {
    window.FB.init({
      appId: '2651158151951093',
      autoLogAppEvents: true,
      xfbml: true,
      version: 'v19.0'
    });
  };
  
  const script = document.createElement('script');
  script.src = 'https://connect.facebook.net/pt_BR/sdk.js';
  script.async = true;
  script.defer = true;
  document.body.appendChild(script);
}, []);

// Launch Embedded Signup
const handleConnectWhatsApp = () => {
  setIsConnecting(true);
  
  window.FB.login(async (response) => {
    if (response.authResponse) {
      const code = response.authResponse.code;
      
      // Meta also sends WABA and phone number via the session info callback
      // We handle this via the "fb_embedded_signup:complete" message event below
    } else {
      setIsConnecting(false);
      toast.error('Conexão cancelada pelo usuário');
    }
  }, {
    config_id: '<YOUR_EMBEDDED_SIGNUP_CONFIG_ID>', // Create this in Meta App Dashboard → WhatsApp → Embedded Signup
    response_type: 'code',
    override_default_response_type: true,
    extras: {
      setup: {},
      featureType: '',
      sessionInfoVersion: '3'
    }
  });
};

// Listen for session info message (contains waba_id and phone_number_id)
useEffect(() => {
  const handleMessage = async (event: MessageEvent) => {
    if (event.origin !== 'https://www.facebook.com' && event.origin !== 'https://web.facebook.com') return;
    
    try {
      const data = JSON.parse(event.data);
      if (data.type === 'WA_EMBEDDED_SIGNUP') {
        if (data.event === 'FINISH') {
          const { phone_number_id, waba_id } = data.data;
          // Now we have everything — exchange the code
          await completeConnection(phone_number_id, waba_id);
        } else if (data.event === 'CANCEL') {
          setIsConnecting(false);
          toast.error('Configuração cancelada');
        } else if (data.event === 'ERROR') {
          setIsConnecting(false);
          toast.error('Erro no processo de conexão Meta');
        }
      }
    } catch {}
  };
  
  window.addEventListener('message', handleMessage);
  return () => window.removeEventListener('message', handleMessage);
}, []);

// Store the code from FB.login callback in a ref, use it in completeConnection
const authCodeRef = useRef<string | null>(null);

const completeConnection = async (phoneNumberId: string, wabaId: string) => {
  try {
    const result = await apiFetch('/api/whatsapp-official/connect', {
      method: 'POST',
      body: JSON.stringify({
        code: authCodeRef.current,
        waba_id: wabaId,
        phone_number_id: phoneNumberId
      })
    });
    
    setConnection(result);
    toast.success('WhatsApp conectado com sucesso! 🎉');
  } catch (err: any) {
    toast.error('Erro ao conectar: ' + err.message);
  } finally {
    setIsConnecting(false);
  }
};
```

## IMPORTANT - Before implementing
In the Meta App Dashboard (developers.facebook.com), you need to:
1. Go to your app → WhatsApp → Configuration → Embedded Signup
2. Create an Embedded Signup configuration
3. Copy the config_id and replace <YOUR_EMBEDDED_SIGNUP_CONFIG_ID> in the code above
4. Set the webhook URL to: https://your-domain.com/api/webhook/meta
5. Set the webhook verify token to match WHATSAPP_WEBHOOK_VERIFY_TOKEN in your .env
6. Subscribe to webhook fields: messages, message_deliveries, message_reads, messaging_postbacks

This must be done BEFORE running this prompt, otherwise the FB.login call will fail.
Document the config_id here so we can hardcode it.
