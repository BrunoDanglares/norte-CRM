function verifyMetaSignature(req: Request, res: Response, next: NextFunction): void {
  const signature = req.headers["x-hub-signature-256"] as string;

  if (!signature || !signature.startsWith("sha256=")) {
    res.sendStatus(403);
    return;
  }

  const appSecret = process.env.WHATSAPP_APP_SECRET;
  if (!appSecret) {
    console.error("[Meta Webhook] WHATSAPP_APP_SECRET não configurado");
    res.sendStatus(500);
    return;
  }

  const expectedSig =
    "sha256=" +
    crypto
      .createHmac("sha256", appSecret)
      .update(JSON.stringify(req.body))
      .digest("hex");

  let sigBuffer: Buffer;
  let expectedBuffer: Buffer;
  try {
    sigBuffer = Buffer.from(signature);
    expectedBuffer = Buffer.from(expectedSig);
  } catch {
    res.sendStatus(403);
    return;
  }

  if (
    sigBuffer.length !== expectedBuffer.length ||
    !crypto.timingSafeEqual(sigBuffer, expectedBuffer)
  ) {
    console.warn("[Meta Webhook] Assinatura inválida recebida");
    res.sendStatus(403);
    return;
  }

  next();
}
