══ comercialAgent.ts — detectSituation() ══

Adicione os casos abaixo ao detectSituation() existente:

  // C2 — Agendamento de instalação
  if (/agendar|agenda|quando instalam|quando vem o técnico|
       marcar instalação|horário de instalação/i.test(texto)) {
    return 'C2';
  }

  // C4 — Consulta de fidelidade e multa
  if (/fidelidade|multa|fidelidade do contrato|

  }

  // C7 — Alteração de titularidade (escalate_always)

  // C9 — Mudança de endereço
  if (/mudar de casa|mudei de endereço|vou mudar|
       transferir minha internet|novo endereço|
       mudança de endereço/i.test(texto)) {
    return 'C9';
  }

  // C10 — Indicação / Referral

       indicar alguém|amigo quer contratar|

══ handle() do ComercialAgent — tratar escalações ══

Após detectar situação, adicione bloco para escalações
obrigatórias (C6 e C7):

  // C6 — Retenção (sempre escala — já deve estar no
  //       CancelamentoAgent, mas pode chegar pelo comercial)
  if (situationCode === 'C6') {
    return {
      message: '',
      redirectTo: 'CANCELAMENTO',
      guardFeedback:
        `Cliente com intenção de cancelamento. ` +
        `Motivo coletado: ${dados.motivo_cancelamento ?? 'não informado'}.`,
    };
  }

  // C7 — Titularidade (sempre escala)

      message: '',

        'requer validação de documentos e alteração contratual.',
    };
  }
