══ MUDANÇA 2 — CRIAR server/services/agents/priorityResolver.ts ══

Implemente o seguinte conteúdo:

  const DEFAULT_PRIORITIES: Record<string, number> = {
    HUMANO: 1,
    CANCELAMENTO: 2,
    FINANCEIRO: 3,
    SUPORTE_TECNICO: 4,
    VENDAS: 5,
    INCIDENTE_REGIONAL: 4,
    GERAL: 6,
  };

  export function resolveIntent(
    vector: IntentVector,
    tenantPriorities?: Record<string, number>
  ): string {
    const priorities = {
      ...DEFAULT_PRIORITIES,
      ...(tenantPriorities ?? {}),
    };

    const candidates = vector.intents
      .filter(i => i.confidence >= 0.3)
      .sort((a, b) => {
        const pa = priorities[a.intent] ?? 99;
        const pb = priorities[b.intent] ?? 99;
        if (pa !== pb) return pa - pb;
        return b.confidence - a.confidence;
      });

    return candidates[0]?.intent ?? vector.dominant;
  }

  export function buildIntentContext(vector: IntentVector): string {
    if (vector.intents.length <= 1) return '';
    const secondary = vector.intents
      .slice(1)
      .map(i => `${i.intent}(${Math.round(i.confidence * 100)}%)`)
      .join(', ');
    return `\nContexto adicional detectado na mensagem: ${secondary}.` +
      ' Use para enriquecer a resposta se relevante.';
  }

  export type { IntentVector };
