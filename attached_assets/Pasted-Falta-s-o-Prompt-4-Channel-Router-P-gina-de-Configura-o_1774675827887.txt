PROMPT 4A — Channel Router (Backend)
Create the channel router service for ChatBanana at server/services/channel-router.ts

This service is the single entry point for ALL outbound WhatsApp messages. It checks if the workspace has an active Meta Cloud API connection and routes through it, otherwise falls back to Z-API.

## File to create: server/services/channel-router.ts

import { db } from '../db';
import { whatsapp_official_connections } from '../db/schema';
import { eq, and } from 'drizzle-orm';
import * as metaWhatsApp from './meta-whatsapp';

export interface SendMessageParams {
  workspaceId: string;
  to: string;
  type: 'text' | 'image' | 'audio' | 'document' | 'video' | 'template';
  content?: string;
  mediaUrl?: string;
  mediaCaption?: string;
  filename?: string;
  templateName?: string;
  templateLanguage?: string;
  templateComponents?: any[];
  replyToMessageId?: string;
  conversationId?: number;
}

export interface SendMessageResult {
  success: boolean;
  messageId?: string;
  channel: 'meta' | 'zapi';
  error?: string;
}

export async function sendMessage(params: SendMessageParams): Promise<SendMessageResult> {
  const metaConnection = await db.query.whatsapp_official_connections.findFirst({
    where: and(
      eq(whatsapp_official_connections.workspace_id, params.workspaceId),
      eq(whatsapp_official_connections.status, 'active')
    )
  });

  if (metaConnection) {
    return sendViaMeta(params, metaConnection);
  }

  return sendViaZAPI(params, params.workspaceId);
}

async function sendViaMeta(params: SendMessageParams, connection: any): Promise<SendMessageResult> {
  try {
    const base = {
      phoneNumberId: connection.phone_number_id,
      accessToken: connection.access_token,
      to: params.to
    };

    let messageId: string;

    switch (params.type) {
      case 'text':
        const textResult = await metaWhatsApp.sendTextMessage({
          ...base,
          text: params.content || '',
          replyToMessageId: params.replyToMessageId
        });
        messageId = textResult.messageId;
        break;

      case 'image':
      case 'audio':
      case 'document':
      case 'video':
        const mediaResult = await metaWhatsApp.sendMediaMessage({
          ...base,
          type: params.type,
          mediaUrl: params.mediaUrl!,
          caption: params.mediaCaption,
          filename: params.filename
        });
        messageId = mediaResult.messageId;
        break;

      case 'template':
        const tplResult = await metaWhatsApp.sendTemplateMessage({
          ...base,
          templateName: params.templateName!,
          language: params.templateLanguage || 'pt_BR',
          components: params.templateComponents || []
        });
        messageId = tplResult.messageId;
        break;

      default:
        throw new Error(`Unsupported message type: ${params.type}`);
    }

    // Update last_used_at non-blocking
    db.update(whatsapp_official_connections)
      .set({ last_used_at: new Date() })
      .where(eq(whatsapp_official_connections.id, connection.id))
      .catch(() => {});

    console.log(`[ChannelRouter] ✅ Sent via Meta | to: ${params.to} | type: ${params.type}`);
    return { success: true, messageId, channel: 'meta' };

  } catch (err: any) {
    console.error('[ChannelRouter] Meta send failed:', err.message);
    return { success: false, error: err.message, channel: 'meta' };
  }
}

  const [showToken, setShowToken] = useState(false);
  const [formData, setFormData] = useState({

    phone_number_id: '',
    access_token: ''

    loadConnection();
  }, []);

  async function loadConnection() {
    try {
      const data = await apiFetch('/api/whatsapp-official/connection');
      setConnection(data.connected === false ? null : data);
      if (data.connected !== false) {
        loadTemplates();
      }
    } catch {
      setConnection(null);
    } finally {
      setLoading(false);
    }
  }

  async function loadTemplates() {
    try {
      const data = await apiFetch('/api/whatsapp-official/templates');
      setTemplates(Array.isArray(data) ? data : []);
    } catch {}
  }

  async function handleConnect() {
    if (!formData.waba_id || !formData.phone_number_id || !formData.access_token) {
      // show toast error: 'Preencha todos os campos'
      return;
    }
    setConnecting(true);
    try {
      const data = await apiFetch('/api/whatsapp-official/connect', {
        method: 'POST',
        body: JSON.stringify({
          waba_id: formData.waba_id,
          phone_number_id: formData.phone_number_id,
          access_token: formData.access_token
        })
      });
      setConnection(data);
      setShowManualForm(false);
      setFormData({ waba_id: '', phone_number_id: '', access_token: '' });
      // toast success: 'WhatsApp conectado com sucesso! 🎉'
      loadTemplates();
    } catch (err: any) {
      // toast error: err.message
    } finally {
      setConnecting(false);
    }
  }

  async function handleTest() {
    setTesting(true);
    try {
      const data = await apiFetch('/api/whatsapp-official/test', { method: 'POST' });
      if (data.success) {
        // toast success: `Conexão OK — ${data.phoneNumber} (${data.qualityRating})`
      } else {
        // toast error: data.error
      }
    } catch (err: any) {
      // toast error: err.message
    } finally {
      setTesting(false);
    }
  }

  async function handleDisconnect() {

      await apiFetch('/api/whatsapp-official/connection', { method: 'DELETE' });
      setConnection(null);
      setTemplates([]);

    } catch (err: any) {
      // toast error: err.message
    }
  }

  async function handleSyncTemplates() {
    setSyncing(true);
    try {
      const data = await apiFetch('/api/whatsapp-official/templates/sync', { method: 'POST' });
      // toast success: `${data.synced} templates sincronizados`
      loadTemplates();
    } catch (err: any) {
      // toast error: err.message
    } finally {
      setSyncing(false);
    }
  }

  function getTierLabel(tier: string) {
    const map: Record<string, string> = {
      TIER_1K: '1.000 msg/dia',
      TIER_10K: '10.000 msg/dia',
      TIER_100K: '100.000 msg/dia',
      TIER_UNLIMITED: 'Ilimitado'

    return map[tier] || tier;

  function getQualityColor(rating: string) {
    if (rating === 'GREEN') return 'text-green-400 bg-green-400/10';
    if (rating === 'YELLOW') return 'text-yellow-400 bg-yellow-400/10';
    return 'text-red-400 bg-red-400/10';

    if (status === 'PENDING') return 'text-yellow-400 bg-yellow-400/10';
    if (status === 'REJECTED') return 'text-red-400 bg-red-400/10';
    return 'text-gray-400 bg-gray-400/10';

        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-purple-500" />
