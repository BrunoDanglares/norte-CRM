### Arquivo 2: `server/services/kanbanArchivalService.ts`

Cria esse arquivo completo:

```typescriptimport { db } from "../db";
import { leads } from "../../shared/schema";
import { eq, and, isNull, inArray } from "drizzle-orm";/**

Stages finais por pipeline — exatamente como estão cadastrados no banco.

COMERCIAL:
"Instalado / Ativado" → ativado
"Perdido"             → perdido

SUPORTE TÉCNICO (3 etapas finais):
"Resolvido"           → ativado
"Visita Técnica"      → ativado
"Escalado / NOC"      → perdido

FINANCEIRO (3 etapas finais):
"Pago / Regularizado" → ativado
"Inadimplente"        → perdido
"Cancelado por Débito"→ perdido
*/
const FINAL_STAGE_MAP: Record<string, "ativado" | "perdido"> = {
// Comercial
"Instalado / Ativado":  "ativado",
"Perdido":              "perdido",
// Suporte Técnico
"Resolvido":            "ativado",
"Visita Técnica":       "ativado",
"Escalado / NOC":       "perdido",
// Financeiro
"Pago / Regularizado":  "ativado",
"Inadimplente":         "perdido",
"Cancelado por Débito": "perdido",
};
const FINAL_STAGE_NAMES = Object.keys(FINAL_STAGE_MAP);/**

Arquiva um lead imediatamente ao ser movido para etapa final.
Chamado pelo frontend/rota quando o lead é arrastado para a coluna final.
*/
export async function archiveLead(
leadId: number,
reason: "ativado" | "perdido"
): Promise<void> {
await db
.update(leads)
.set({
archivedAt: new Date(),
archivalReason: reason,
})
.where(eq(leads.id, leadId));
}
/**

Detecta automaticamente o motivo pelo nome do stage.
Útil quando o arquivamento vem de uma movimentação de stage sem reason explícito.
*/
export function resolveArchivalReason(stageName: string): "ativado" | "perdido" {
return FINAL_STAGE_MAP[stageName] ?? "perdido";
}
/**

Job de limpeza de fim de expediente.
Arquiva todos os leads nas etapas finais que ainda não foram arquivados.
Deve ser chamado às 18h (seg-sex) e 12h (sáb).
*/
export async function archiveEndOfShift(): Promise<{ archived: number }> {
const leadsToArchive = await db
.select({ id: leads.id, stage: leads.stage })
.from(leads)
.where(
and(
isNull(leads.archivedAt),
inArray(leads.stage, FINAL_STAGE_NAMES)
)
);
if (leadsToArchive.length === 0) return { archived: 0 };for (const lead of leadsToArchive) {
const reason = resolveArchivalReason(lead.stage ?? "");
await db
.update(leads)
.set({ archivedAt: new Date(), archivalReason: reason })
.where(eq(leads.id, lead.id));
}console.log([KanbanArchival] ${leadsToArchive.length} leads arquivados (fim de expediente));
return { archived: leadsToArchive.length };
}
