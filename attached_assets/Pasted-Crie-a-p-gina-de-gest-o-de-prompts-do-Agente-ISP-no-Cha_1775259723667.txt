import { useState, useEffect } from 'react';
import { useToast } from '@/hooks/use-toast';
import { apiFetch } from '@/lib/apiFetch';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { Badge } from '@/components/ui/badge';
import { RefreshCw, Save, Bot } from 'lucide-react';

const AGENTS = [
  { name: 'agente_financeiro',         label: 'Financeiro',         desc: 'Cobranças, 2ª via, liberação 48h' },
  { name: 'agente_suporte_online',     label: 'Suporte Online',     desc: 'Cliente conectado com problemas' },
  { name: 'agente_suporte_offline',    label: 'Suporte Offline',    desc: 'Cliente sem internet' },
  { name: 'agente_incidente_regional', label: 'Incidente Regional', desc: 'Problema coletivo de rede' },
  { name: 'agente_cancelamento',       label: 'Cancelamento',       desc: 'Retenção de clientes' },
  { name: 'agente_vendas',             label: 'Vendas',             desc: 'Novos clientes e upgrades' },
  { name: 'cliente_nao_encontrado',    label: 'Não Encontrado',     desc: 'Cliente sem cadastro no ERP' },
];

export default function ISPPromptsPage() {
  const { toast } = useToast();
  const [templates, setTemplates] = useState<Record<string, string>>({});
  const [saving, setSaving] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);
  const [active, setActive] = useState(AGENTS[0].name);

  useEffect(() => { loadTemplates(); }, []);

  async function loadTemplates() {
    setLoading(true);
    try {
      const res = await apiFetch('/api/n8n/ai/templates');
      const data = await res.json();
      const map: Record<string, string> = {};
      (data.templates ?? []).forEach((t: any) => {
        map[t.name] = t.system_prompt;
      });
      setTemplates(map);
    } catch {
      toast({ title: 'Erro ao carregar templates', variant: 'destructive' });
    } finally {
      setLoading(false);
    }
  }

  async function saveTemplate(name: string) {
    setSaving(name);
    try {
      const res = await apiFetch(`/api/n8n/ai/templates/${name}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ system_prompt: templates[name] }),
      });
      if (!res.ok) throw new Error();
      toast({ title: 'Prompt salvo!' });
    } catch {
      toast({ title: 'Erro ao salvar', variant: 'destructive' });
    } finally {
      setSaving(null);
    }
  }

  async function seedTemplates() {
    try {
      await apiFetch('/api/n8n/ai/seed', { method: 'POST' });
      await loadTemplates();
      toast({ title: 'Templates padrão restaurados!' });
    } catch {
      toast({ title: 'Erro ao restaurar', variant: 'destructive' });
    }
  }

  const agent = AGENTS.find(a => a.name === active)!;

  return (
    <div className="p-6 max-w-5xl mx-auto">
      <div className="flex items-center justify-between mb-6">
        <div className="flex items-center gap-3">
          <Bot className="text-purple-400" size={22} />
          <div>
            <h1 className="text-lg font-semibold text-white">Agentes ISP</h1>
            <p className="text-xs text-gray-400">Edite os prompts. Mudanças valem imediatamente.</p>
          </div>
        </div>
        <Button variant="outline" size="sm" onClick={seedTemplates} className="text-xs">
          <RefreshCw size={12} className="mr-1" /> Restaurar padrões
        </Button>
      </div>

      <div className="flex gap-6">
        {/* Sidebar */}
        <div className="w-48 flex-shrink-0 space-y-1">
          {AGENTS.map(a => (
            <button
              key={a.name}
              onClick={() => setActive(a.name)}
              className={`w-full text-left px-3 py-2 rounded-lg text-sm transition-colors ${
                active === a.name
                  ? 'bg-purple-600 text-white'
                  : 'text-gray-400 hover:bg-white/5 hover:text-white'
              }`}
            >
              {a.label}
            </button>
          ))}
        </div>

        {/* Editor */}
        <div className="flex-1 rounded-xl border border-white/10 bg-[#1c1830] p-5">
          {loading ? (
            <div className="flex items-center justify-center h-64">
              <RefreshCw className="animate-spin text-purple-400" size={20} />
            </div>
          ) : (
            <>
              <div className="flex items-center justify-between mb-4">
                <div>
                  <h2 className="font-medium text-white">{agent.label}</h2>
                  <p className="text-xs text-gray-400 mt-0.5">{agent.desc}</p>
                </div>
                <Badge variant="outline" className="text-purple-400 border-purple-400/30 text-xs font-mono">
                  {agent.name}
                </Badge>
              </div>

              <Textarea
                value={templates[agent.name] ?? ''}
                onChange={e => setTemplates(prev => ({ ...prev, [agent.name]: e.target.value }))}
                placeholder="Escreva o prompt do agente aqui..."
                className="min-h-[360px] font-mono text-sm bg-[#0f0c1a] border-white/10 text-gray-200 resize-y"
              />

              <div className="flex justify-end mt-4">
                <Button
                  onClick={() => saveTemplate(agent.name)}
                  disabled={saving === agent.name}
                  className="bg-purple-600 hover:bg-purple-700 text-white text-sm"
                >
                  {saving === agent.name
                    ? <RefreshCw size={13} className="mr-2 animate-spin" />
                    : <Save size={13} className="mr-2" />
                  }
                  Salvar prompt
                </Button>
              </div>
            </>
          )}
        </div>
      </div>
    </div>
  );
}
